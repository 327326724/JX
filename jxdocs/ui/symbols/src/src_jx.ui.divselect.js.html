  7 Jet().$package(function (J) {
  8     var packageContext = this,
  9         $D = J.dom,
 10         $S = J.string,
 11         $E = J.event;
 12     J.ui = J.ui || {};
 13     /*
 14        * div模式select相关，可以移植到其他地方 示例： html:<div id="testdiv"></div> js:
 15        * divSelect.show("testdiv","test",selOptions,"",160,"test()");
 16        * 
 17        */
 18     /**
 19      * 
 20      * div模拟的下拉框, 直接 new一个实例即可
 21      * 
 22      * @memberOf ui
 23      * @name DivSelect
 24      * @class
 25      * @constructor
 26      * @param {String} objId 容器
 27      * @param {String} selectName 名称
 28      * @param {Object} dataObj 数据[数组]
 29      * @param {Object} selOption 默认项
 30      * @param {Number} width 宽度
 31      * @param {Boolean} isUpper
 32      * 
 33      * @example
 34      * html: <div id="testdiv"></div> 
 35      * js: new DivSelect("testdiv","test",selOptions,"",160);
 36      * 
 37      */
 38      var DivSelect = J.ui.DivSelect = new J.Class(
 39 
 40      {
 41         /*
 42          * 主调函数 参数含义：容器，名称，数据[数组]，默认项，宽度 注意：数据格式
 43          */
 44         /** @ignore */
 45         init : function(objId, selectName, dataObj, selOption, width, isUpper) {
 46 
 47           this._selectName = selectName;
 48           this._dataObj = dataObj;
 49           this._selOption = selOption;
 50           this._isUpper = isUpper || false;
 51 
 52           // _divSelect = divSelect; ;
 53           var data = dataObj.data;
 54           var _Obj = document.getElementById(objId);
 55           if (!_Obj || typeof(_Obj) === "undefined") {
 56             return false;
 57           }
 58           var s1 = document.createElement("div");
 59           if (isNaN(width) || width == "") {
 60             width = 150;
 61           } else if (width < 26) {
 62             width = 26;
 63           }
 64           s1.style.width = width;
 65           var top = ""
 66           if(this._isUpper){
 67               s1.style.position = "relative";
 68               var height = (15*(data.length)); //每个item 15px
 69               top = "top:-"+((height>150?150:height)+4)+"px;";
 70           }
 71 
 72           var str = "";
 73           /*
 74            * //判断是否有数据
 75            */
 76           // 判断是否有数据
 77           if (data.length > 0) {
 78             // 有数据时显示数据选项列表
 79             str += "<input type='hidden' name='" + selectName
 80                 + "' id='" + selectName + "' value='"
 81                 + this.relv(selOption, data) + "'>";
 82             str += "<div id='_a_" + selectName + "' style='width:"
 83                 + width
 84                 + "px;height:18px; border:1px #728EA4 solid;'>";
 85 
 86             str += "<div id='_v_"
 87                 + selectName
 88                 + "' style='position:relative;float:left;left:2px;width:"
 89                 + (width - 22)
 90                 + "px;height:18px;font-size:12px;overflow:hidden;line-height:18px;'>"
 91                 + this.reStr(data, selOption) + "</div>";
 92             str += "<div id='_arr_"
 93                 + selectName
 94                 + "' class='divSelect_arr' style='position:relative;float:right;right:0px;width:18px;height:18px;text-align:center;font-family:Webdings;font-size:16px;overflow:hidden;background-color:#FFF;cursor:pointer!important;cursor:hand;'></div>";
 95             str += "</div>";
 96             str += "<div id='_b_"
 97                 + selectName
 98                 + "' style='position:absolute; background-color:#FFFFFF; width:"
 99                 + width
100                 + "px; height:"
101                 + this.height(data.length)
102                 + "px;border:1px #728EA4 solid;overflow-x:hidden;overflow-y:auto;display:none; z-index:99999;" + top + "'>";
103             for (var i = 0; i < data.length; i++) {
104               var style = data[i][0] == selOption
105                   ? this.style(2)
106                   : this.style(1);
107               str += "<div id='_s_" + selectName + i
108                   + "' style='" + style + "' >" + data[i][1]
109                   + "</div>";
110             }
111             str += "</div>";
112           } else {
113             // 没有数据时显示一个空窗体
114             str += "<input type='hidden' name='" + selectName
115                 + "' id='" + selectName + "' value='"
116                 + selOption + "'>";
117             str += "<div id='_a_" + selectName + "' style='width:"
118                 + width
119                 + "px;height:18px; border:1px #728EA4 solid;'>";
120             str += "<div id='_v_"
121                 + selectName
122                 + "' style='position:relative;float:left;left:2px;width:"
123                 + (width - 22)
124                 + "px;height:18px;font-size:12px;overflow:hidden;line-height:18px;'></div>";
125             str += "<div id='_arr_"
126                 + selectName
127                 + "' class='divSelect_arr' style='position:relative;float:right;right:0px;width:18px;height:18px;text-align:center;font-family:Webdings;font-size:16px;overflow:hidden;background-color:#FFF;cursor:pointer!important;cursor:hand;'></div>";
128             str += "</div>";
129             str += "<div id='_b_"
130                 + selectName
131                 + "' style='position:absolute; background-color:#FFFFFF; width:"
132                 + width
133                 + "px; height:"
134                 + this.height(0)
135                 + "px;border:1px #728EA4 solid;overflow-x:hidden;overflow-y:auto;display:none; z-index:99999;'></div>";
136           }
137 
138           s1.innerHTML = str;
139           _Obj.appendChild(s1);
140 
141           var divSelectObj = this;
142 
143           var clickShowOption = function() {
144             divSelectObj.showOptions();
145           }
146 
147           $E.on($D.id('_v_' + selectName), "click", clickShowOption);
148           $E.on($D.id('_arr_' + selectName), "click",
149                   clickShowOption);
150           if (data.length > 0) {
151             var optionOver = function() {
152               divSelectObj.css(this, 3);
153             };
154             var optionOut = function() {
155               divSelectObj.css(this, 1);
156             };
157             var optionClick = function() {
158               divSelectObj.selected(this);
159             };
160             for (var i = 0; i < data.length; i++) {
161               $E.on($D.id('_s_' + selectName + i), "mouseover",
162                   optionOver);
163               $E.on($D.id('_s_' + selectName + i), "mouseout",
164                   optionOut);
165               $E.on($D.id('_s_' + selectName + i), "click",
166                   optionClick);
167             }
168 
169           }
170         },
171         /*
172          * clickShowOption : function(){ divSelectObj.showOptions(); },
173          */
174         /**
175          * 返回选定项的值
176          * @param {String} n id
177          * @return {String}
178          */
179         value : function(n) {
180           n = n || this._selectName;
181           return document.getElementById(n).value;
182         },
183         /**
184          * 返回选定项的文本
185          * @param {String} n id
186          * @return {String}
187          */
188         text : function(n) {
189           n = n || this._selectName;
190           return document.getElementById("_v_" + n).innerHTML;
191         },
192         /**
193          * 选中给节点
194          * @param {HTMLElement} optionObj 
195          */
196         selected : function(optionObj) {
197           var data = this._dataObj.data;
198           var value = optionObj.innerHTML;
199           for (var i = 0; i < data.length; i++) {
200             if (data[i][1] === value) {
201               $D.id(this._selectName).value = data[i][0];
202             }
203             $D.id('_s_' + this._selectName + i).style.cssText = this.style(1);
204           }
205 
206           $D.id("_v_" + this._selectName).innerHTML = value;
207 
208           optionObj.style.cssText = this.style(2);
209           this.hidden();
210           $E.notifyObservers(this, "selectedChanged");
211         },
212         relv : function(v, d) {
213           for (var i = 0; i < d.length; i++) {
214             if (d[i][0] == v) {
215               return v;
216             }
217           }
218           if (v == null || v == "") {
219             return d[0][0];
220           }
221         },
222         reStr : function(d, m) {
223           for (var i = 0; i < d.length; i++) {
224             if (d[i][0] == m) {
225               return d[i][1];
226             }
227           }
228           if (m == null || m == "") {
229             return d[0][1];
230           }
231         },
232         /**
233          * 获取列表的高度
234          * @return {Number}
235          */
236         height : function(l) {
237           var h;
238           if (l > 10 || l < 1)
239             h = 10 * 15;
240           else
241             h = l * 15;
242           h += 2;
243           return h;
244         },
245         showOptions : function(n) {
246           n = n || this._selectName;
247           var o = document.getElementById("_b_" + n)
248           if (o.style.display == "none")
249             o.style.display = "";
250           else
251             o.style.display = "none";
252 
253         },
254         hidden : function() {
255           document.getElementById("_b_" + this._selectName).style.display = "none";
256         },
257         style : function(m) {
258           var cs = "";
259           switch (m) {
260             case 1 :
261               cs = "height:15px; font-size:12px; line-height:15px; overflow:hidden; background-color:#FFFFFF; color:#000000; font-weight:normal;";
262               break;
263             case 2 :
264               cs = "height:15px; font-size:12px; line-height:15px; overflow:hidden; background-color:#315DAD; color:#FFFFFF; font-weight:bold;";
265               break;
266             case 3 : //mouseover
267               cs = "height:15px; font-size:12px; line-height:15px; overflow:hidden; background-color:#D8D8D8; color:#000000; font-weight:normal;";
268               break;
269           }
270           return cs;
271         },
272         css : function(optionObj, type) {
273           if (type === 1) {
274             if ($D.id("_v_" + this._selectName).innerHTML != optionObj.innerHTML) {
275               optionObj.style.cssText = this.style(type);
276             }
277           } else {
278             if ($D.id("_v_" + this._selectName).innerHTML != optionObj.innerHTML) {
279               optionObj.style.cssText = this.style(type);
280             }
281           }
282         }
283       });
284       
285 });
