1 Jx().$package(function (J) {
  2     var $D = J.dom,
  3         $E = J.event;
  4         
  5     J.ui = J.ui || {};
  6 
  7     /**
  8      * 【Pagination】
  9      * 
 10      * @class 
 11      * @memberOf ui
 12      * @name Pagination
 13      * @param {Object} option 参数对象 option:{el:DOM分页容器, count: 总数, prePage: 每页数量, type: 分页类型(1/2), callback: 回调}<br/>
 14      * option.type 分页类型,type=1:带省略号,有页码列表;type=2:无省略号,没页码列表
 15      * @since version 1.0
 16      * @description Jx.ui.Pagination是一个分页组件，当在页面上需要使用列表来展示较多数据时，可以使用该组件实现分页切换效果。
 17      */
 18      J.ui.Pagination = new J.Class(
 19      /**
 20      * @lends ui.Pagination.prototype
 21      */
 22      {
 23         /**
 24         * @ignore
 25         */
 26         init:function(option){
 27             this.pagebarEl = option.el;
 28             this.count = option.count;
 29             this.perPage = option.perPage || 10;
 30             this.callback = option.callback||function(){};
 31             this.type = option.type||1;
 32             this.maxPage = Math.ceil(this.count/this.perPage);
 33             this.currentPage=option.currentPage||1;
 34             var paginationEl = this.pagebarEl.children[0];
 35             this.pagelistEl = paginationEl.children[1];
 36             this.preEl = paginationEl.children[0];
 37             this.nextEl = paginationEl.children[2];
 38             
 39             
 40             if (this.maxPage > 1) {
 41                 $D.show(this.pagebarEl);
 42             } else {
 43                 $D.hide(this.pagebarEl);
 44             }
 45             //this.setPage(this.currentPage);
 46             this.turnTo(this.currentPage, true);
 47             $E.on(this.preEl,"click",J.bind(this.onPreElClick,this));
 48             $E.on(this.nextEl,"click",J.bind(this.onNextElClick,this));
 49         },
 50         
 51         /**
 52         * 重置分页组件
 53         * @param {Object} option 参数对象 option:{el:DOM分页容器, count: 总数, prePage: 每页数量, type: 分页类型(1/2), callback: 回调}
 54         * @return
 55         */
 56         reset : function(option){
 57             this.count = J.isNumber(option.count)?option.count:this.count;
 58             this.perPage = option.perPage ||this.perPage|| 10;
 59             this.callback = option.callback||this.callback||function(){};
 60             this.type = option.type||this.type||1;
 61             this.currentPage = option.currentPage||this.currentPage||1;
 62             this.maxPage = Math.ceil(this.count/this.perPage);
 63             if (this.maxPage > 1) {
 64                 $D.show(this.pagebarEl);
 65             } else {
 66                 $D.hide(this.pagebarEl);
 67             }
 68             this.setPage(this.currentPage);
 69         },
 70         
 71         /**
 72         * @private
 73         * 设置
 74         */
 75         setPage:function(page){
 76             var html = "", totalPage = this.maxPage;
 77             this.currentPage = page;
 78 
 79             if (page > totalPage || page < 1) {
 80                 return;
 81             }
 82             //第一种模版,带省略号
 83             if(this.type == 1){
 84                 if( totalPage > 8 ){
 85                     html += '<a class="w'+ ((page+'').length) + (page==1?' cur':"")+'" href="#" >1</a>';
 86                     if(page < 4){
 87                         for(var i=2; i<5; i++){
 88                             html += '<a class="w'+ ((i+'').length) +(i==page?' cur':"")+'" href="#" >'+i+'</a>';
 89                         }
 90                         html += '<span>...</span>';
 91                     }else if( (page+3) > totalPage){
 92                         html += '<span>...</span>';  
 93                         for(var i=(totalPage-3); i<totalPage; i++){
 94                             html += '<a class="w'+ ((i+'').length) +(i==page?' cur':"")+'" href="#" >'+i+'</a>';
 95                         }
 96                     }else{ 
 97                         html += '<span>...</span>';
 98                         for(var i=(page-1); i < (page+2); i++){
 99                             html += '<a class="w'+ ((i+'').length) +(i==page?' cur':"")+'" href="#" >'+i+'</a>';
100                         }
101                         html += '<span>...</span>';
102                     }
103                     html += '<a class="w'+ ((totalPage+'').length) +(page==totalPage?' cur':"")+'" href="#" >'+totalPage+'</a>';
104                 }else{
105                     for( var i=1; i<=totalPage; i++){
106                         html += '<a class="w'+ ((i+'').length) +(i==page?' cur':"")+'" href="#" >'+i+'</a>';
107                     }
108                 }
109     
110                 this.pagelistEl.innerHTML = html;
111                 $D.setStyle(this.pagebarEl,"width","auto");
112                 $D.setStyle(this.pagebarEl,"width",$D.getClientWidth(this.pagebarEl.children[0])+10+"px");
113                 var pagelink = $D.tagName("a",this.pagelistEl), i=0, len=pagelink.length;
114                 for (; i<len; i++) {
115                     $E.on(pagelink[i],"click",J.bind(this.onPagesElClick,this));
116                 }
117                 
118             }else if(this.type==2){//第二种模版,不带省略号
119                 html = page+"/"+totalPage;
120                 this.pagelistEl.innerHTML = html;
121             }
122             
123 
124             $D.removeClass(this.preEl,"preBtn_dis");
125             $D.removeClass(this.nextEl,"nextBtn_dis");
126 
127             if(page == 1){
128                 $D.addClass(this.preEl,"preBtn_dis");
129             } else if(page == totalPage){
130                 $D.addClass(this.nextEl,"nextBtn_dis");
131             }
132         },
133         
134         /**
135         * 切换到 指定页
136         * @param page {Number} 页码
137         * @param force {Boolean} 内部使用参数，可以不填或false
138         */
139         turnTo: function(page, force){
140             if (isNaN(page) || page < 1 || page > this.maxPage || (!force && page == this.currentPage)) {
141                 return;
142             }
143 
144             this.setPage(page);
145             this.callback({start:this.getStart(), end:this.getEnd()});
146         },
147         
148         /**
149         * @private 
150         */
151         onPagesElClick:function(e){
152             e.stopPropagation();
153             e.preventDefault();
154 
155             var page = +e.currentTarget.innerHTML;
156             this.turnTo(page);
157             /*if (isNaN(page) || page < 1 || page > this.maxPage || page == this.currentPage) {
158                 return;
159             }
160 
161             this.setPage(page);
162             this.callback({start:this.getStart(), end:this.getEnd()});*/
163             //$E.notifyObservers(this.context, "SetPage", {start:this.getStart(), end:this.getEnd()});
164         },
165         
166         /**
167         * @private 
168         */
169         onPreElClick:function(e){
170             e.stopPropagation();
171             e.preventDefault();
172 
173             if (this.currentPage > 1) {
174                 this.currentPage--;
175                 this.setPage(this.currentPage);
176                 this.callback({start:this.getStart(), end:this.getEnd()});
177                 //$E.notifyObservers(this.context, "SetPage", {start:this.getStart(), end:this.getEnd()});
178             }
179         },
180         
181         /**
182         * @private 
183         */
184         onNextElClick:function(e){
185             e.stopPropagation();
186             e.preventDefault();
187 
188             if (this.currentPage < this.maxPage) {
189                 this.currentPage++;
190                 this.setPage(this.currentPage);
191                 this.callback({start:this.getStart(), end:this.getEnd()});
192                 //$E.notifyObservers(this.context, "SetPage", {start:this.getStart(), end:this.getEnd()});
193             }
194         },
195         
196         /**
197         * 获取当前页列表起始索引值
198         * @param
199         * @return {Number} 当前页列表起始索引值
200         */
201         getStart:function(){
202             return (this.currentPage-1)*this.perPage;
203         },
204         
205         /**
206         * 获取当前页列表结束索引值
207         * @param
208         * @return {Number} 当前页列表结束索引值
209         */
210         getEnd:function(){
211             var end = (this.currentPage*this.perPage)-1;
212             if(end > this.count - 1)
213                 end = this.count - 1;
214             return end;
215         }
216     });
217 });
