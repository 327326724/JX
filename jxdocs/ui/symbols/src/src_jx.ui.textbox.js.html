  8 Jet().$package(function (J) {
  9     var $D = J.dom,
 10         $S = J.string,
 11         $E = J.event;
 12     
 13     J.ui = J.ui || {};
 14     
 15     /**
 16      * 【TextBox】
 17      * 
 18      * @class 
 19      * @name TextBox
 20      * @memberOf ui
 21      * @constructor
 22      * @param {Object} option 构造参数
 23      * @example
 24      * {
 25      * id: '',//random
 26      * name: id,
 27      * container: document.body,
 28      * className: '' ,
 29      * title: '',
 30      * text: '',
 31      * height: 200,
 32      * width: 200,
 33      * limit: 200,
 34      * readOnly: false,
 35      * hintLink: [{
 36      *  text:'',
 37      *  click:''
 38      *  }]
 39      * }
 40      * 
 41      * var sharebox = new J.ui.TextBox({
 42      *      title: '分享',
 43      *      text: '内容',
 44      *      hint: [{
 45      *          text: '更多',
 46      *          click: ''                   
 47      *      }]
 48      *  });
 49      *  sharebox.show(200, 200);
 50      * @since version 1.0
 51      * @description
 52      */
 53     var TextBox = J.ui.TextBox= new J.Class(
 54     /**
 55      * @lends ui.TextBox.prototype
 56      */
 57     {
 58         /**
 59          * @ignore
 60          */
 61         init: function(option){
 62             var id = 'share_box_' + (option.id || (new Date().getTime()));
 63             var name = option.name || id;
 64             var parent = option.container || document.body;
 65             var className = option.className || '';
 66             var content = option.text || '';
 67             this._titleHeight = 22;
 68             this._footerHeight = 26;
 69             this._padding = 4 * 2;
 70             this._margin = 4 * 2;
 71             option.width = option.width || 200;
 72             option.height = option.height || 100;
 73             option.limit = option.limit || 0;
 74             option.isPopup = option.isPopup || false;
 75             option.readOnly = option.readOnly || false;
 76             option.maskIframe = option.maskIframe || false;
 77             option.hasCloseButton = option.hasCloseButton === false ? false : true;
 78             this._isShow = false;
 79             
 80             var container = this._el = $D.node('div', {
 81                 id: id,
 82                 'class': 'share_box ' + className
 83             });
 84                 
 85             var html = '\
 86                 <div class="share_box_title" id="'+ id +'_title">\
 87                     <div class="share_box_titleTxt" id="'+ id +'_titleTxt"></div>\
 88                 </div>\
 89                 <div class="share_box_body" id="'+ id +'_body">\
 90                     <!--textarea class="share_box_text" id="'+ id +'_text"></textarea-->\
 91                 </div>\
 92                 <div class="share_box_footer" id="'+ id +'_footer">\
 93                     <div class="share_box_showthumb" id="'+id+'_showthumb"></div>\
 94                     <div class="share_box_hint" id="'+ id +'_hint"></div>\
 95                     <div class="share_box_count" id="'+ id +'_count"></div>\
 96                 </div>\
 97                 <div>\
 98                     <img id="'+id+'_thumb" class="share_box_thumb" src="'+option.thumb+'" width=211 height=127 />\
 99                 </div>';
100             
101             if(option.maskIframe){
102                 html += '<iframe class="ui_maskBgIframe" src="'+alloy.CONST.MAIN_URL+'domain.html;" border="0"></iframe>';
103             }
104             
105             container.innerHTML = html;
106             parent.appendChild(container);
107             if(option.isPopup){
108                 this._panel = new J.ui.PopupBox({
109                     id: id,
110                     name: name,
111                     container: container
112                 });
113             }else{
114                 this._panel = new J.ui.Panel({
115                     id: id,
116                     name: name,
117                     container: container
118                 });
119             }
120             
121             
122             this._shareBoxTitleTxt = $D.id(id + '_titleTxt');
123             this._shareBoxTitle = $D.id(id + '_title');
124             this._shareBoxHint = $D.id(id + '_hint');
125             this._shareBoxFooter = $D.id(id + '_footer');
126             this._shareBoxBody = $D.id(id + '_body');
127             this._shareBoxCount = $D.id(id + '_count');
128             this._shareBoxShowthumb=$D.id(id + '_showthumb');
129             this._shareBoxThumb=$D.id(id + '_thumb');
130             
131             var areaHtml = '<strong class="share_big_quote share_left_quote">“</strong><textarea class="share_box_text" id="'+ id +'_text"></textarea><strong class="share_big_quote share_right_quote">”</strong>';
132             if(option.readOnly){
133                 //this._shareBoxText.readOnly = 'readonly';
134                 areaHtml = '<strong class="share_big_quote share_left_quote">“</strong><textarea class="share_box_text" id="'+ id +'_text" readOnly="readonly"></textarea><strong class="share_big_quote share_right_quote">”</strong>';
135             }
136             if(!option.thumb){
137                 $D.hide(this._shareBoxShowthumb);
138             }
139             this._shareBoxBody.innerHTML = areaHtml;
140             this._shareBoxText = $D.id(id + '_text');
141             
142             var context = this;
143             var observer = {
144                 /**
145                  * @ignore
146                  */
147                 onSendButtonClick: function(e){
148                     e.preventDefault();
149                     e.stopPropagation();
150                     $E.notifyObservers(context, 'clickSendButton');
151                 },
152                 /**
153                  * @ignore
154                  */
155                 onCloseButtonClick: function(e){
156                     e.preventDefault();
157                     e.stopPropagation();
158                     context.hide();
159                     $E.notifyObservers(context, 'clickCloseButton');
160                 },
161                 /**
162                  * @ignore
163                  */
164                 onTextAreaKeyUp: function(e){
165                     e.stopPropagation();
166                     e.preventDefault();
167                     alloy.util.subStringByChar(e, option.limit);
168                 },
169                 /**
170                  * @ignore
171                  */
172                 toggleThumb: function(e){
173                     e.preventDefault();
174                     e.stopPropagation();
175                     if($D.isShow(context._shareBoxThumb)){
176                         $D.hide(context._shareBoxThumb);
177                     }
178                     else{
179                         $D.show(context._shareBoxThumb);
180                     }
181                 }
182             };
183             
184             this._sendButton = new J.ui.Button({
185                 'appendTo' : this._shareBoxFooter,
186                 'className' : 'window_button window_ok',
187                 'isStopPropagation' : true,
188                 'text' : '发表' ,
189                 'event' : {
190                     'click' : observer.onSendButtonClick
191                 }
192             });
193             this._sendButton.show();
194             if(option.hasCloseButton){
195                 this._closeButton = new J.ui.Button({
196                     'appendTo' : this._shareBoxTitle,
197                     'className' : 'textbox_button',
198                     'isStopPropagation' : true,
199                     'title' : '关闭' ,
200                     'event' : {
201                         'click' : observer.onCloseButtonClick
202                     }
203                 });
204                 this._closeButton.show();
205             }
206             
207             this._shareBoxText.innerHTML = content;
208             
209             if(option.title){
210                 this._shareBoxTitleTxt.innerHTML = option.title;
211                 $D.show(this._shareBoxTitle);
212             }
213             if(option.hint){
214                 var spliter = '',
215                     node = null;
216                 
217                 for(var i=0; i<option.hint.length; i++){
218                     node = $D.node('a', {
219                         'href': '###',
220                         'class': 'share_box_hintLink'
221                     });
222                     node.innerHTML = option.hint[i].text;
223                     $E.on(node, 'click', option.hint[i].click);
224                     this._shareBoxHint.appendChild(node);
225                 }
226             }
227             if(option.limit){
228                 //暂时去掉数字提示
229 //              this._shareBoxCount.innerHTML = option.maxChar;
230 //              $D.show(this._shareBoxCount);
231                 $E.on(this._shareBoxText, 'keyup', observer.onTextAreaKeyUp);
232             }
233 //            if(option.onTextAreaClick){
234 //                $E.on(this._shareBoxText, 'keyup', option.onTextAreaClick);
235 //            }
236             $E.on(this._shareBoxShowthumb, 'click', observer.toggleThumb);
237             
238             this.setHeight(option.height);
239             this.setWidth(option.width);
240         },
241         /**
242          * 设置高度
243          * @param {Number} height
244          */
245         setHeight: function(height){
246             var h = height - this._titleHeight - this._footerHeight;
247             $D.setStyle(this._shareBoxText, 'height', h + 'px');
248         },
249         /**
250          * 设置宽度
251          * @param {Number} width
252          */
253         setWidth: function(width){
254             $D.setStyle(this._el, 'width', width + 'px');
255             $D.setStyle(this._shareBoxText, 'width', (width - this._padding - this._margin) + 'px');
256         },
257         /**
258          * @private
259          */
260         addMask: function(el){
261             if(!J.browser.ie) 
262                 return;
263             
264             if(J.isString(el)){
265                 var iframe = '<iframe class="ui_maskBgIframe" src="'+alloy.CONST.MAIN_URL+'domain.html"></iframe>';
266                 return el+iframe;
267             }else{
268                 var iframe = $D.node('iframe',{
269                     'class': 'ui_maskBgIframe',
270                     'src': 'about:blank'
271                 }); 
272                 el.appendChild(iframe);
273             }
274         },
275         /**
276          * 获取textbox所处节点
277          */
278         getElement: function(){
279             return this._el;
280         },
281         /**
282          * 返回是否显示
283          */
284         isShow: function(){
285             return this._isShow;    
286         },
287         /**
288          * 在指定位置显示textbox
289          * @param {Number} x
290          * @param {Number} y
291          */
292         show: function(x, y){
293             if(x && y){
294                 this._panel.setXY(x, y);
295             }
296             this._panel.show();
297             this._isShow = true;
298         },
299         /**
300          * 隐藏
301          */
302         hide: function(){
303             this._panel.hide();
304             this._isShow = false;
305             $E.notifyObservers(this, 'hide');
306         },
307         /**
308          * 获取用户输入的内容
309          */
310         getValue: function(){
311             return this._shareBoxText.value;
312         },
313         /**
314          * 设置输入框的内容
315          */
316         setValue: function(value){
317             this._shareBoxText.value = value;
318         },
319         /**
320          * 给状态栏设置提醒节点
321          * @private
322          */
323         setHint: function(el){
324             this._shareBoxHint.innerHTML = '';
325             this._shareBoxHint.appendChild(el);
326             $D.show(this._shareBoxHint);
327         },
328         /**
329          * @private
330          */
331         setThumb: function(url){
332             this._shareBoxThumb.src=url;
333         }
334     });
335 });
