1 Jx().$package(function(J){
  2     var $D = J.dom,
  3         $E = J.event;
  4     
  5     /**
  6      * iPad滚动模拟
  7      * 
  8      * @memberOf ui
  9      * @class
 10      * @name IframeScroller
 11      * @constructor
 12      * 
 13      * @param {HTMLElement} iframe
 14      * 
 15      */
 16     var IframeScroller = function(iframe){
 17         this.container=iframe.parentNode;
 18         if(J.platform.iPad && J.platform.iPad.split(".")[0]>=4) {
 19             this.container=iframe.parentNode.parentNode;
 20         }
 21         //console.info("container width:"+$D.getWidth(this.container));
 22         this.iframe=iframe;
 23         this.holding=false;
 24         this.posx=0;
 25         this.posy=0;
 26         this.offsetX=0;
 27         this.offsetY=0;
 28         var context = this;
 29         /** @ignore */
 30         this.observers={
 31             /** @ignore */
 32             onTouchStart:function(e){
 33                 var touch = e.changedTouches[0];
 34                 context.posx = touch.pageX;
 35                 context.posy = touch.pageY;
 36                 //console.info("iframe height: "+ $D.getHeight(context.iframe));
 37                 context.minX = $D.getWidth(context.container) - $D.getWidth(context.iframe);
 38                 context.minY = $D.getHeight(context.container) - $D.getHeight(context.iframe);
 39                 
 40                 //console.info("minx:"+context.minX+",miny:"+context.minY);
 41                 
 42                 $E.on(context.iframe,"touchmove",context.observers.onTouchMove);
 43                 $E.on(context.iframe,"touchend",context.observers.onTouchEnd);
 44             },
 45             /** @ignore */
 46             onTouchMove:function(e){
 47                 if(e.changedTouches.length>1)return;
 48                 e.preventDefault();
 49                 e.stopPropagation();
 50                 var touch = e.changedTouches[0];
 51                 var posx = touch.pageX;
 52                 var posy = touch.pageY;
 53                 var dx = context.posx - posx;
 54                 var dy = context.posy - posy;               
 55                 
 56                 var newpx = context.offsetX - dx;
 57                 var newpy = context.offsetY - dy;
 58                 
 59                 //console.info("iframe touch move:"+newpx+","+newpy);
 60                 
 61                 if(newpx > 0) newpx = 0;
 62                 else if(newpx < context.minX) newpx = context.minX;
 63                 if(newpy > 0) newpy = 0;
 64                 else if(newpy < context.minY) newpy = context.minY;
 65                 
 66                 //console.info("apple sucks: "+newpy);
 67                 //console.info("iframe touch move:"+newpx+","+newpy);
 68                 
 69                 iframe.style['left'] = newpx + 'px';
 70                 iframe.style['top'] = newpy + 'px';
 71                 
 72             
 73                 context.offsetX = newpx;
 74                 context.offsetY = newpy;
 75                 context.posx = posx;
 76                 context.posy = posy;
 77             },
 78             /** @ignore */
 79             onTouchEnd:function(){
 80                 $E.off(context.iframe,"touchmove",context.observers.onTouchMove);
 81                 $E.off(context.iframe,"touchend",context.observers.onTouchEnd);
 82             }
 83         };
 84         /**
 85          * 销毁
 86          * @name destroy
 87          * @function
 88          * @memberOf ui.IframeScroller.prototype
 89          */
 90         this.destroy=function(){
 91             $E.off(this.iframe,"touchstart",this.observers.onTouchStart);   
 92             $E.off(this.iframe,"touchmove",this.observers.onTouchMove);
 93             $E.off(this.iframe,"touchend",this.observers.onTouchEnd);
 94             this.iframe=null;
 95             this.container=null;
 96         }
 97         $E.on(this.iframe,"touchstart",this.observers.onTouchStart);            
 98     }
 99     J.ui = J.ui || {};
100     J.ui.IframeScroller = IframeScroller;
101 });
