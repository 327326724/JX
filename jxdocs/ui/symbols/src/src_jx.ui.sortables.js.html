 64         refreshDropTarget : function(target){
 65             var targetPos,
 66                 xy,
 67                 targetWidth,
 68                 dropTargets = this.dropTargets,
 69                 targetHeight;
 70             this.dropPos = [];
 71             if(!target){
 72                 for(var i in dropTargets){
 73                     target = dropTargets[i].el;
 74                     targetPos = {};
 75                     xy = $D.getXY(target);
 76                     targetWidth = $D.getClientWidth(target);
 77                     targetHeight = $D.getClientHeight(target);
 78                     targetPos.x = xy[0];
 79                     targetPos.y = xy[1];
 80                     targetPos.w = targetWidth;
 81                     targetPos.h = targetHeight;
 82                     this.dropPos[i] = targetPos;
 83                 }
 84             }else{
 85                 for(var i in dropTargets){
 86                     dropTargetEl = dropTargets[i].el;
 87                     if(target.el === dropTargetEl){
 88                         targetPos = {};
 89                         xy = $D.getXY(dropTargetEl);
 90                         targetWidth = $D.getClientWidth(dropTargetEl);
 91                         targetHeight = $D.getClientHeight(dropTargetEl);
 92                         targetPos.x = xy[0];
 93                         targetPos.y = xy[1];
 94                         targetPos.w = targetWidth;
 95                         targetPos.h = targetHeight;
 96                         this.dropPos[i] = targetPos;
 97                         break;
 98                     }
 99                 }
100             }

102         /**

104          * @param {HTMLElement} el
105          */
106         addDragClass : function(el){
107             var parentEl = el.parentNode, 
108                 apperceiveEl = el,
109 //              effectEl =this.effectEl||this.clone(apperceiveEl),//转移到beforeStart里创建
110                 effectEl = false,
111                 context = this,
112                 currentDropTarget = null,
113                 xyP = $D.getXY(parentEl),//父节点
114                 xyS = $D.getXY(apperceiveEl),
115                 addId = el.getAttribute(this.sortStr)||"",
116                 dropTargets = this.dropTargets,
117                 nextEl,
118                 preXY;//子节点
119             
120 //          //获取模型的体积
121 //          var _width = $D.getClientWidth(apperceiveEl);
122 //          var _height = $D.getClientHeight(apperceiveEl);
123 //          var margin ={}; 
124 //              margin.top= parseInt($D.getStyle(apperceiveEl,"marginTop")||0);
125 //              margin.buttom= parseInt($D.getStyle(apperceiveEl,"marginbottom")||0);
126 //              margin.left= parseInt($D.getStyle(apperceiveEl,"marginLeft")||0);
127 //              margin.right= parseInt($D.getStyle(apperceiveEl,"marginRight")||0);
128 //          //修正盒子模型
129 //          _width += (margin.left+margin.right);
130 //          _height += (margin.top+margin.buttom);
131 
132             this.dropPos = [];
133             var onDragBeforeStart = function(){
134                 _width = $D.getClientWidth(apperceiveEl);
135                 _height = $D.getClientHeight(apperceiveEl);
136                 var margin ={}; 
137                 margin.top= parseInt($D.getStyle(apperceiveEl,"marginTop")||0);
138                 margin.buttom= parseInt($D.getStyle(apperceiveEl,"marginbottom")||0);
139                 margin.left= parseInt($D.getStyle(apperceiveEl,"marginLeft")||0);
140                 margin.right= parseInt($D.getStyle(apperceiveEl,"marginRight")||0);
141                 //修正盒子模型
142                 _width += (margin.left+margin.right);
143                 _height += (margin.top+margin.buttom);
144                 //重新赋值,因为dropTarget是动态添加的
145                 dropTargets = this.dropTargets;
146                 //az 2011-6-8
147                 effectEl = context.effectEl || context.clone(apperceiveEl);
148                 context.dragController[addId].setEffect(effectEl);
149                 //重新设置XY.这里不能在dragstart设置,因为在发出start之前已经用了这个高度
150                 setStartXY();
151                 this.refreshDropTarget();
152                 //储存下个一个节点,以便恢复
153                 parentEl = apperceiveEl.parentNode;
154                 nextEl = apperceiveEl.nextSibling;
155                 
156                 document.body.appendChild(effectEl);
157                 
158                 $E.notifyObservers(context, "beforeStart");
159             };
160             var onDragStart = function(option){
161                 currentDropTarget = null;
162                 if(context.option.isDash){
163                     apperceiveEl = context.cloneDashedEl(apperceiveEl);
164                 }else{
165                     if(!J.browser.ie){
166                         //$D.setStyle(apperceiveEl,"opacity",0.5);
167                         $D.addClass(apperceiveEl,"ui_halfOpacity");
168                     }
169                 }
170                 //J.out("drag开始");
171                 $E.notifyObservers(context, "start");
172             };
173             var onDragMove = function(option){
174                 var moveEvent = option.orientEvent;
175                 var moveX = moveEvent.pageX;
176                 var moveY = moveEvent.pageY;
177                 //减少计算
178                 if(Math.abs(moveX-preXY[0])+Math.abs(moveY-preXY[1])<1){
179                     return;
180                 }else{
181                     if(moveX-preXY[0]>0){
182                         var direction = "right";
183                     }else{
184                         var direction = "left";
185                     }
186                     preXY = [moveX,moveY];
187                 }
188                 //父节点宽度
189                 var hitTarget = {el:null,level:-1,index:0};
190                 for(var i in this.dropPos){
191                     if(
192                         (moveX > this.dropPos[i].x)&&
193                         (moveX < this.dropPos[i].x + this.dropPos[i].w)&&
194                         (moveY > this.dropPos[i].y)&&
195                         (moveY < this.dropPos[i].y + this.dropPos[i].h)
196                         ){
197                             var dropTargetObj = dropTargets[i];
198                             var dropTarget = dropTargetObj.el;
199                             var parentWidth = $D.getClientWidth(dropTarget);
200                             //每行个数
201                             var perRow = Math.floor(parentWidth/_width);
202                             
203                             //容器的XY
204                             var dropTargetXY = $D.getXY(dropTarget);
205                             //先对于容器的位置
206                             var x = moveX - dropTargetXY[0];
207                             var y = moveY - dropTargetXY[1];
208                             //var fixX = preIndexXY[1]==indexY?(preXY[0]-moveX>0?(-0.3):0.3):0;
209                             var indexY = Math.floor(y/_height);
210                             if(direction=="right"){
211                                 var indexX = Math.ceil(x/_width);
212                             }else if(direction=="left"){
213                                 var indexX = Math.floor(x/_width);
214                             }
215                             
216                             var index = indexX+indexY*perRow;
217                             if(hitTarget.level <dropTargetObj.level){
218                                 hitTarget.level = dropTargetObj.level;
219                                 hitTarget.el = dropTargetObj.el;
220                                 hitTarget.index = index;
221                             }
222                             /*
223                             if(dropTarget.getAttribute('customAcceptDrop')){
224                                 $E.notifyObservers(dropTarget,'dragmove',option);
225                                 break;
226                             }
227                             
228                             
229                             
230                             //修正IE下index找不到的时候插不到最后一个的错误
231                             if(dropTarget.childNodes[index]){
232                                 dropTarget.insertBefore(apperceiveEl,dropTarget.childNodes[index]);
233                             }else{
234                                 dropTarget.appendChild(apperceiveEl);
235                             }
236                             break;
237                             */
238                     }
239                 }
240                 currentDropTarget = hitTarget.el
241                 if(currentDropTarget){
242                     if(currentDropTarget.getAttribute('customAcceptDrop')){
243                         $E.notifyObservers(currentDropTarget,'dragmove',option);
244                     }else if(currentDropTarget.childNodes[index]){//修正IE下index找不到的时候插不到最后一个的错误
245                         currentDropTarget.insertBefore(apperceiveEl,currentDropTarget.childNodes[index]);
246                     }else{
247                         currentDropTarget.appendChild(apperceiveEl);
248                     }
249                 }
250 //              try{
251                     $E.notifyObservers(context, "move",option);
252 //              }catch(e){
253 //                  J.out("move error");
254 //              }
255             };
256             var onDragOverFlow = function(option){
257                 option.el=effectEl;
258                 $E.notifyObservers(context, "overFlowBorder",option);
259             };
260             var onDragEnd = function(option){
261                 var moveEvent = option.orientEvent;
262                 var moveX = moveEvent.pageX;
263                 var moveY = moveEvent.pageY;
264                 if(currentDropTarget && currentDropTarget.getAttribute &&currentDropTarget.getAttribute('customAcceptDrop')){
265                     $E.notifyObservers(currentDropTarget, "drop",{'dragEl':el,"queue":queue,"pos":{x:moveX,y:moveY},"apperceiveEl":apperceiveEl,"nextEl":nextEl,"parentEl":parentEl,"currentDropTarget":currentDropTarget});
266                 }
267                 document.body.removeChild(effectEl);
268                 if(context.option.isDash){
269                     context.removeDashedEl();
270                     apperceiveEl = context.tempEl;
271                 }else{
272                     if(!J.browser.ie){
273                         //$D.setStyle(apperceiveEl,"opacity",1);
274                         $D.removeClass(apperceiveEl,"ui_halfOpacity");
275                     }                   
276                 }
277                 //返回队列
278                 var queue = [];
279                 for(var i in dropTargets){
280                     queue[i] = [];
281                     var childNodes = dropTargets[i].el.childNodes;
282                     for(var k=0;k<childNodes.length;k++){
283                         var node = childNodes[k];
284                         if(!node.getAttribute){
285                             break;
286                         }
287                         var attribute=childNodes[k].getAttribute(context.sortStr);
288                         if(attribute){
289                             queue[i].push(attribute);
290                         }
291                     }
292                 }
293                 //J.out("drag结束");
294                 try{
295                     $E.notifyObservers(context, "end",{"queue":queue,"pos":option,"apperceiveEl":apperceiveEl,"nextEl":nextEl,"parentEl":parentEl});
296                 }catch(e){
297                     J.out("drop error");
298                 }
299                 var containerEl=document.elementFromPoint(moveX, moveY);
300                 if(containerEl){
301                     $E.notifyObservers(J.ui, "drop",{'dragEl':el,"pos":{x:moveX,y:moveY},"apperceiveEl":apperceiveEl,"dropEl":containerEl});

353         /**

356         cloneDashedEl : function(el){
357             var dashedEl = $D.node("div");
358             var className = this.option.className;
359             if(className){
360                 $D.setClass(dashedEl,className);
361             }else{
362                 $D.setStyle(dashedEl,"border","dashed 2px #fff");
363                 $D.setClass(dashedEl,el.className);
364                 $D.setStyle(dashedEl,"position","relative");
365                 $D.setStyle(dashedEl,"float","left");
366                 var width = el.offsetWidth - 10 * parseInt(dashedEl.style.borderWidth) + "px";
367                 var height = el.offsetHeight - 10 * parseInt(dashedEl.style.borderWidth) + "px"; 
368                 $D.setStyle(dashedEl,"width",width);
369                 $D.setStyle(dashedEl,"height",height);
370             }
371             this.dashedEl = dashedEl;
372             if (el.nextSibling) {
373                 el.parentNode.insertBefore(dashedEl, el.nextSibling);
374             }
375             else {
376                 el.parentNode.appendChild(dashedEl);
377             }
378             this.tempEl = el;
379             el.parentNode.removeChild(el);
380             return dashedEl;
381         },
382         /**
383          * @private
384          */
385         removeDashedEl : function(){
386             if (this.dashedEl.nextSibling) {
387                 this.dashedEl.parentNode.insertBefore(this.tempEl, this.dashedEl.nextSibling);
388             }
389             else {
390                 this.dashedEl.parentNode.appendChild(this.tempEl);
391             }
392             this.dashedEl.parentNode.removeChild(this.dashedEl);
393         },

434         forEachNode : function(arr,fun){
435             var len = arr.length;
436             if (typeof fun != "function") {
437                 throw new TypeError();
438             }
439             var thisp = arguments[2];
440             for (var i = 0; i < len; i++) {
441                 if (i in arr) {
442                     fun.call(thisp, arr[i], i, arr);
443                 }
444             }
445         }
