  2 Jx().$package(function (J) {
  3     var $D = J.dom,
  4         $E = J.event;
  5 
  6 
  7     J.ui = J.ui || {};
  8     /**
  9      * 列表滚动类
 10      * @class 
 11      * @memberOf ui
 12      * @name Marquee
 13      * @constructor
 14      * @param {Object} option
 15      * option.speed || 40;//滚动速度
 16        option.stopTime || 3000;//滚动时长
 17        option.lineHeight || 20;//列表每行的高度
 18        target;//需要滚动的列表
 19      * @since version 1.0
 20      */
 21     J.ui.Marquee = new J.Class(
 22     /**
 23      * @lends ui.Marquee.prototype
 24      */
 25     {
 26         /**
 27          * @ignore
 28          */
 29         init : function(option){
 30             var marqueeContext = this;
 31             this.speed = option.speed || 40;
 32             this.stopTime = option.stopTime || 3000;
 33             this.lineHeight = option.lineHeight || 20;
 34             this.target = option.target;
 35             this.timer = null;
 36             this.lineTimer = null;
 37             this.intervaler = null;
 38             this.scrollHeight = this.lineHeight;
 39             this.isStop = false;
 40             
 41             this._onTimeRun = function(){
 42                 marqueeContext.scrollOneLine();
 43             };
 44     
 45         },
 46         /**
 47          * 滚动一行
 48          */
 49         scrollOneLine : function(){
 50             if (this.scrollHeight > 0) {
 51                 this.scrollHeight--;
 52                 var currentTop = this.target.style.top.match(/-?\d+/);
 53                 currentTop = (!currentTop) ? 0 : parseInt(currentTop[0]);
 54                 this.target.style.top = (--currentTop) + 'px';
 55                 
 56 
 57                 this.lineTimer = setTimeout(this._onTimeRun, this.speed);
 58             }else{
 59                 if(!this.isStop){
 60                     this.update();
 61                 }
 62             }
 63 
 64         },
 65         /**
 66          * 停止当前行滚动
 67          */
 68         stop : function() {
 69             if (this.timer) {
 70                 clearTimeout(this.timer);
 71             }
 72         },
 73         /**
 74          * 完全停止滚动
 75          */
 76         stopAll : function(){
 77             this.stop();
 78             if(this.lineTimer){
 79                 clearTimeout(this.lineTimer);
 80             }
 81         },
 82         /**
 83          * 重设列表的位置
 84          */
 85         reset : function(){
 86             this.target.style.top = '0px';
 87         },
 88         /**
 89          * 执行一次滚动
 90          */
 91         update : function(){
 92             if(this.isStop){
 93                 return;
 94             }
 95             if (this.timer) {
 96                 clearTimeout(this.timer);
 97             }
 98             this.scrollHeight = this.lineHeight;
 99             var currentTop = this.target.style.top.match(/\d+/);
100             var height = $D.getScrollHeight(this.target);
101             if(!!currentTop && !!height){
102                 currentTop = parseInt(currentTop[0]);
103                 if(currentTop >= height){
104                     this.target.style.top = this.lineHeight + 'px';
105                     this.scrollOneLine();
106                     return;
107                 }
108             }
109 
110             this.timer = setTimeout(this._onTimeRun, this.stopTime);
111         },
112         /**
113          * 继续执行上一次未完成的滚动
114          */
115         walkOnLastLine : function(){
116             this._onTimeRun();
117         }
118         
119     });
120 
121 
122 
123 });
