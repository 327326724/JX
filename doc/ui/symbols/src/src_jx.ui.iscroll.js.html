 17 /**
 18  * iPad 的自定义滚动条
 19  * @class 
 20  * @memberOf ui
 21  * @name iScroll
 22  * @constructor
 23  * @param {HTMLElement} el
 24  * @param {Object} options
 25  * 
 26  * @since version 1.0
 27  */
 28 function iScroll (el, options) {
 29     var that = this, i;
 30     that.element = typeof el == 'object' ? el : document.getElementById(el);
 31     that.wrapper = that.element.parentNode;
 32 
 33     that.element.style.webkitTransitionProperty = '-webkit-transform';
 34     that.element.style.webkitTransitionTimingFunction = 'cubic-bezier(0,0,0.25,1)';
 35     that.element.style.webkitTransitionDuration = '0';
 36     that.element.style.webkitTransform = translateOpen + '0,0' + translateClose;
 37 
 38     // Default options
 39     that.options = {
 40         bounce: has3d,
 41         momentum: has3d,
 42         checkDOMChanges: true,
 43         topOnDOMChanges: false,
 44         hScrollbar: has3d,
 45         vScrollbar: has3d,
 46         fadeScrollbar: isIthing || !isTouch,
 47         shrinkScrollbar: isIthing || !isTouch,
 48         desktopCompatibility: false,
 49         overflow: 'auto',
 50         snap: false,
 51         bounceLock: false,
 52         scrollbarColor: 'rgba(0,0,0,0.5)',
 53         /** @ignore */
 54         onScrollEnd: function () {}
 55     };
 56     
 57     // User defined options
 58     if (typeof options == 'object') {
 59         for (i in options) {
 60             that.options[i] = options[i];
 61         }
 62     }
 63 
 64     if (that.options.desktopCompatibility) {
 65         that.options.overflow = 'hidden';
 66     }
 67     
 68     that.onScrollEnd = that.options.onScrollEnd;
 69     delete that.options.onScrollEnd;
 70     
 71     that.wrapper.style.overflow = that.options.overflow;
 72     
 73     that.refresh();
 74 
 75     window.addEventListener('onorientationchange' in window ? 'orientationchange' : 'resize', that, false);
 76 
 77     if (isTouch || that.options.desktopCompatibility) {
 78         that.element.addEventListener(START_EVENT, that, false);
 79         that.element.addEventListener(MOVE_EVENT, that, false);
 80         that.element.addEventListener(END_EVENT, that, false);
 81     }
 82     
 83     if (that.options.checkDOMChanges) {
 84         that.element.addEventListener('DOMSubtreeModified', that, false);
 85     }
 86 }

 95     y: 0,

 98     handleEvent: function (e) {
 99         var that = this;
100         
101         switch (e.type) {
102             case START_EVENT:
103                 that.touchStart(e);
104                 break;
105             case MOVE_EVENT:
106                 that.touchMove(e);
107                 break;
108             case END_EVENT:
109                 that.touchEnd(e);
110                 break;
111             case 'webkitTransitionEnd':
112                 that.transitionEnd();
113                 break;
114             case 'orientationchange':
115             case 'resize':
116                 that.refresh();
117                 break;
118             case 'DOMSubtreeModified':
119                 that.onDOMModified(e);
120                 break;
121         }
122     },
123     /** @ignore */
124     onDOMModified: function (e) {
125         var that = this;
126 
127         // (Hopefully) execute onDOMModified only once
128         if (e.target.parentNode != that.element) {
129             return;
130         }
131 
132         setTimeout(function () { that.refresh(); }, 0);
133 
134         if (that.options.topOnDOMChanges && (that.x!=0 || that.y!=0)) {
135             that.scrollTo(0,0,'0');
136         }

138     /**
139      * 刷新滚动条
140      */
141     refresh: function () {
142         var that = this,
143             resetX = that.x, resetY = that.y,
144             snap;
145         
146         that.scrollWidth = that.wrapper.clientWidth;
147         that.scrollHeight = that.wrapper.clientHeight;
148         that.scrollerWidth = that.element.offsetWidth;
149         that.scrollerHeight = that.element.offsetHeight;
150         that.maxScrollX = that.scrollWidth - that.scrollerWidth;
151         that.maxScrollY = that.scrollHeight - that.scrollerHeight;
152         that.directionX = 0;
153         that.directionY = 0;
154 
155         if (that.scrollX) {
156             if (that.maxScrollX >= 0) {
157                 resetX = 0;
158             } else if (that.x < that.maxScrollX) {
159                 resetX = that.maxScrollX;
160             }
161         }
162         if (that.scrollY) {
163             if (that.maxScrollY >= 0) {
164                 resetY = 0;
165             } else if (that.y < that.maxScrollY) {
166                 resetY = that.maxScrollY;
167             }
168         }
169 
170         // Snap
171         if (that.options.snap) {
172             that.maxPageX = -Math.floor(that.maxScrollX/that.scrollWidth);
173             that.maxPageY = -Math.floor(that.maxScrollY/that.scrollHeight);
174 
175             snap = that.snap(resetX, resetY);
176             resetX = snap.x;
177             resetY = snap.y;
178         }
179 
180         if (resetX!=that.x || resetY!=that.y) {
181             that.setTransitionTime('0');
182             that.setPosition(resetX, resetY, true);
183         }
184         
185         that.scrollX = that.scrollerWidth > that.scrollWidth;
186         that.scrollY = !that.options.bounceLock && !that.scrollX || that.scrollerHeight > that.scrollHeight;
187 
188         // Update horizontal scrollbar
189         if (that.options.hScrollbar && that.scrollX) {
190             that.scrollBarX = that.scrollBarX || new scrollbar('horizontal', that.wrapper, that.options.fadeScrollbar, that.options.shrinkScrollbar, that.options.scrollbarColor);
191             that.scrollBarX.init(that.scrollWidth, that.scrollerWidth);
192         } else if (that.scrollBarX) {
193             that.scrollBarX = that.scrollBarX.remove();
194         }
195 
196         // Update vertical scrollbar
197         if (that.options.vScrollbar && that.scrollY && that.scrollerHeight > that.scrollHeight) {
198             that.scrollBarY = that.scrollBarY || new scrollbar('vertical', that.wrapper, that.options.fadeScrollbar, that.options.shrinkScrollbar, that.options.scrollbarColor);
199             that.scrollBarY.init(that.scrollHeight, that.scrollerHeight);
200         } else if (that.scrollBarY) {
201             that.scrollBarY = that.scrollBarY.remove();
202         }
203     },
204     /** @ignore */
205     setPosition: function (x, y, hideScrollBars) {
206         var that = this;
207         
208         that.x = x;
209         that.y = y;
210 
211         that.element.style.webkitTransform = translateOpen + that.x + 'px,' + that.y + 'px' + translateClose;
212 
213         // Move the scrollbars
214         if (!hideScrollBars) {
215             if (that.scrollBarX) {
216                 that.scrollBarX.setPosition(that.x);
217             }
218             if (that.scrollBarY) {
219                 that.scrollBarY.setPosition(that.y);
220             }
221         }
222     },

230     /**
231      * 设置滚动的动画时间
232      * @param {Number} time
233      */
234     setTransitionTime: function(time) {
235         var that = this;
236         
237         time = time || '0';
238         that.element.style.webkitTransitionDuration = time;
239         
240         if (that.scrollBarX) {
241             that.scrollBarX.bar.style.webkitTransitionDuration = time;
242             that.scrollBarX.wrapper.style.webkitTransitionDuration = has3d && that.options.fadeScrollbar ? '300ms' : '0';
243         }
244         if (that.scrollBarY) {
245             that.scrollBarY.bar.style.webkitTransitionDuration = time;
246             that.scrollBarY.wrapper.style.webkitTransitionDuration = has3d && that.options.fadeScrollbar ? '300ms' : '0';
247         }
248     },
249     /** @ignore */
250     touchStart: function(e) {
251         var that = this,
252             matrix;
253         
254         if (!that.enabled) {
255             return;
256         }
257 
258         e.preventDefault();
259         e.stopPropagation();
260         
261         that.scrolling = true;      // This is probably not needed, but may be useful if iScroll is used in conjuction with other frameworks
262 
263         that.moved = false;
264         that.distX = 0;
265         that.distY = 0;
266 
267         that.setTransitionTime('0');
268 
269         // Check if the scroller is really where it should be
270         if (that.options.momentum || that.options.snap) {
271             matrix = new WebKitCSSMatrix(window.getComputedStyle(that.element).webkitTransform);
272             if (matrix.e != that.x || matrix.f != that.y) {
273                 document.removeEventListener('webkitTransitionEnd', that, false);
274                 that.setPosition(matrix.e, matrix.f);
275                 that.moved = true;
276             }
277         }
278 
279         that.touchStartX = isTouch ? e.changedTouches[0].pageX : e.pageX;
280         that.scrollStartX = that.x;
281 
282         that.touchStartY = isTouch ? e.changedTouches[0].pageY : e.pageY;
283         that.scrollStartY = that.y;
284 
285         that.scrollStartTime = e.timeStamp;
286 
287         that.directionX = 0;
288         that.directionY = 0;
289     },
290     /** @ignore */
291     touchMove: function(e) {
292         if (!this.scrolling) {
293             return;
294         }
295 
296         var that = this,
297             pageX = isTouch ? e.changedTouches[0].pageX : e.pageX,
298             pageY = isTouch ? e.changedTouches[0].pageY : e.pageY,
299             leftDelta = that.scrollX ? pageX - that.touchStartX : 0,
300             topDelta = that.scrollY ? pageY - that.touchStartY : 0,
301             newX = that.x + leftDelta,
302             newY = that.y + topDelta;
303 
304         //e.preventDefault();
305         e.stopPropagation();    // Stopping propagation just saves some cpu cycles (I presume)
306 
307         that.touchStartX = pageX;
308         that.touchStartY = pageY;
309 
310         // Slow down if outside of the boundaries
311         if (newX >= 0 || newX < that.maxScrollX) {
312             newX = that.options.bounce ? Math.round(that.x + leftDelta / 3) : (newX >= 0 || that.maxScrollX>=0) ? 0 : that.maxScrollX;
313         }
314         if (newY >= 0 || newY < that.maxScrollY) { 
315             newY = that.options.bounce ? Math.round(that.y + topDelta / 3) : (newY >= 0 || that.maxScrollY>=0) ? 0 : that.maxScrollY;
316         }
317 
318         if (that.distX + that.distY > 5) {          // 5 pixels threshold
319 
320             // Lock scroll direction
321             if (that.distX-3 > that.distY) {
322                 newY = that.y;
323                 topDelta = 0;
324             } else if (that.distY-3 > that.distX) {
325                 newX = that.x;
326                 leftDelta = 0;
327             }
328 
329             that.setPosition(newX, newY);
330             that.moved = true;
331             that.directionX = leftDelta > 0 ? -1 : 1;
332             that.directionY = topDelta > 0 ? -1 : 1;
333         } else {
334             that.distX+= Math.abs(leftDelta);
335             that.distY+= Math.abs(topDelta);
336 //          that.dist+= Math.abs(leftDelta) + Math.abs(topDelta);
337         }
338     },
339     /** @ignore */
340     touchEnd: function(e) {
341         if (!this.scrolling) {
342             return;
343         }
344 
345         var that = this,
346             time = e.timeStamp - that.scrollStartTime,
347             point = isTouch ? e.changedTouches[0] : e,
348             target, ev,
349             momentumX, momentumY,
350             newDuration = 0,
351             newPositionX = that.x, newPositionY = that.y,
352             snap;
353 
354         that.scrolling = false;
355 
356         if (!that.moved) {
357             that.resetPosition();
358 
359             if (isTouch) {
360                 // Find the last touched element
361                 target = point.target;
362                 while (target.nodeType != 1) {
363                     target = target.parentNode;
364                 }
365 
366                 // Create the fake event
367                 ev = document.createEvent('MouseEvents');
368                 ev.initMouseEvent('click', true, true, e.view, 1,
369                     point.screenX, point.screenY, point.clientX, point.clientY,
370                     e.ctrlKey, e.altKey, e.shiftKey, e.metaKey,
371                     0, null);
372                 ev._fake = true;
373                 target.dispatchEvent(ev);
374             }
375 
376             return;
377         }
378 
379         if (!that.options.snap && time > 250) {         // Prevent slingshot effect
380             that.resetPosition();
381             return;
382         }
383 
384         if (that.options.momentum) {
385             momentumX = that.scrollX === true
386                 ? that.momentum(that.x - that.scrollStartX,
387                                 time,
388                                 that.options.bounce ? -that.x + that.scrollWidth/5 : -that.x,
389                                 that.options.bounce ? that.x + that.scrollerWidth - that.scrollWidth + that.scrollWidth/5 : that.x + that.scrollerWidth - that.scrollWidth)
390                 : { dist: 0, time: 0 };
391 
392             momentumY = that.scrollY === true
393                 ? that.momentum(that.y - that.scrollStartY,
394                                 time,
395                                 that.options.bounce ? -that.y + that.scrollHeight/5 : -that.y,
396                                 that.options.bounce ? (that.maxScrollY < 0 ? that.y + that.scrollerHeight - that.scrollHeight : 0) + that.scrollHeight/5 : that.y + that.scrollerHeight - that.scrollHeight)
397                 : { dist: 0, time: 0 };
398 
399             newDuration = Math.max(Math.max(momentumX.time, momentumY.time), 1);        // The minimum animation length must be 1ms
400             newPositionX = that.x + momentumX.dist;
401             newPositionY = that.y + momentumY.dist;
402         }
403 
404         if (that.options.snap) {
405             snap = that.snap(newPositionX, newPositionY);
406             newPositionX = snap.x;
407             newPositionY = snap.y;
408             newDuration = Math.max(snap.time, newDuration);
409         }
410 
411         that.scrollTo(newPositionX, newPositionY, newDuration + 'ms');
412     },
413     /** @ignore */
414     transitionEnd: function () {
415         var that = this;
416         document.removeEventListener('webkitTransitionEnd', that, false);
417         that.resetPosition();
418     },
419     /**
420      * 重置滚动条的位置
421      */
422     resetPosition: function () {
423         var that = this,
424             resetX = that.x,
425             resetY = that.y;
426 
427         if (that.x >= 0) {
428             resetX = 0;
429         } else if (that.x < that.maxScrollX) {
430             resetX = that.maxScrollX;
431         }
432 
433         if (that.y >= 0 || that.maxScrollY > 0) {
434             resetY = 0;
435         } else if (that.y < that.maxScrollY) {
436             resetY = that.maxScrollY;
437         }
438         
439         if (resetX != that.x || resetY != that.y) {
440             that.scrollTo(resetX, resetY);
441         } else {
442             if (that.moved) {
443                 that.onScrollEnd();     // Execute custom code on scroll end
444                 that.moved = false;
445             }
446 
447             // Hide the scrollbars
448             if (that.scrollBarX) {
449                 that.scrollBarX.hide();
450             }
451             if (that.scrollBarY) {
452                 that.scrollBarY.hide();
453             }
454         }
455     },
456     
457     snap: function (x, y) {
458         var that = this, time;
459 
460         if (that.directionX > 0) {
461             x = Math.floor(x/that.scrollWidth);
462         } else if (that.directionX < 0) {
463             x = Math.ceil(x/that.scrollWidth);
464         } else {
465             x = Math.round(x/that.scrollWidth);
466         }
467         that.pageX = -x;
468         x = x * that.scrollWidth;
469         if (x > 0) {
470             x = that.pageX = 0;
471         } else if (x < that.maxScrollX) {
472             that.pageX = that.maxPageX;
473             x = that.maxScrollX;
474         }
475 
476         if (that.directionY > 0) {
477             y = Math.floor(y/that.scrollHeight);
478         } else if (that.directionY < 0) {
479             y = Math.ceil(y/that.scrollHeight);
480         } else {
481             y = Math.round(y/that.scrollHeight);
482         }
483         that.pageY = -y;
484         y = y * that.scrollHeight;
485         if (y > 0) {
486             y = that.pageY = 0;
487         } else if (y < that.maxScrollY) {
488             that.pageY = that.maxPageY;
489             y = that.maxScrollY;
490         }
491 
492         // Snap with constant speed (proportional duration)
493         time = Math.round(Math.max(
494                 Math.abs(that.x - x) / that.scrollWidth * 500,
495                 Math.abs(that.y - y) / that.scrollHeight * 500
496             ));
497             
498         return { x: x, y: y, time: time };
499     },
500     /**
501      * 滚动到指定位置
502      * @param {Number} destX
503      * @param {Number} destY
504      * @param {Number} runtime 滚动动画的执行时间
505      */
506     scrollTo: function (destX, destY, runtime) {
507         var that = this;
508 
509         if (that.x == destX && that.y == destY) {
510             that.resetPosition();
511             return;
512         }
513 
514         that.moved = true;
515         that.setTransitionTime(runtime || '350ms');
516         that.setPosition(destX, destY);
517 
518         if (runtime==='0' || runtime=='0s' || runtime=='0ms') {
519             that.resetPosition();
520         } else {
521             document.addEventListener('webkitTransitionEnd', that, false);  // At the end of the transition check if we are still inside of the boundaries
522         }
523     },
524     /**
525      * 按页滚动
526      * @param {Number} pageX
527      * @param {Number} pageY
528      * @param {Number} runtime 滚动动画的执行时间
529      */
530     scrollToPage: function (pageX, pageY, runtime) {
531         var that = this, snap;
532 
533         if (!that.options.snap) {
534             that.pageX = -Math.round(that.x / that.scrollWidth);
535             that.pageY = -Math.round(that.y / that.scrollHeight);
536         }
537 
538         if (pageX == 'next') {
539             pageX = ++that.pageX;
540         } else if (pageX == 'prev') {
541             pageX = --that.pageX;
542         }
543 
544         if (pageY == 'next') {
545             pageY = ++that.pageY;
546         } else if (pageY == 'prev') {
547             pageY = --that.pageY;
548         }
549 
550         pageX = -pageX*that.scrollWidth;
551         pageY = -pageY*that.scrollHeight;
552 
553         snap = that.snap(pageX, pageY);
554         pageX = snap.x;
555         pageY = snap.y;
556 
557         that.scrollTo(pageX, pageY, runtime || '500ms');
558     },
559     /**
560      * 滚动到指定节点处
561      * @param {HTMLElement} el
562      * @param {Number} runtime 滚动动画的执行时间
563      */
564     scrollToElement: function (el, runtime) {
565         el = typeof el == 'object' ? el : this.element.querySelector(el);
566 
567         if (!el) {
568             return;
569         }
570 
571         var that = this,
572             x = that.scrollX ? -el.offsetLeft : 0,
573             y = that.scrollY ? -el.offsetTop : 0;
574 
575         if (x >= 0) {
576             x = 0;
577         } else if (x < that.maxScrollX) {
578             x = that.maxScrollX;
579         }
580 
581         if (y >= 0) {
582             y = 0;
583         } else if (y < that.maxScrollY) {
584             y = that.maxScrollY;
585         }
586 
587         that.scrollTo(x, y, runtime);
588     },
589     
590     momentum: function (dist, time, maxDistUpper, maxDistLower) {
591         var friction = 2.5,
592             deceleration = 1.2,
593             speed = Math.abs(dist) / time * 1000,
594             newDist = speed * speed / friction / 1000,
595             newTime = 0;
596 
597         // Proportinally reduce speed if we are outside of the boundaries 
598         if (dist > 0 && newDist > maxDistUpper) {
599             speed = speed * maxDistUpper / newDist / friction;
600             newDist = maxDistUpper;
601         } else if (dist < 0 && newDist > maxDistLower) {
602             speed = speed * maxDistLower / newDist / friction;
603             newDist = maxDistLower;
604         }
605         
606         newDist = newDist * (dist < 0 ? -1 : 1);
607         newTime = speed / deceleration;
608 
609         return { dist: Math.round(newDist), time: Math.round(newTime) };
610     },
611     /**
612      * 销毁
613      */
614     destroy: function (full) {
615         var that = this;
616 
617         window.removeEventListener('onorientationchange' in window ? 'orientationchange' : 'resize', that, false);      
618         that.element.removeEventListener(START_EVENT, that, false);
619         that.element.removeEventListener(MOVE_EVENT, that, false);
620         that.element.removeEventListener(END_EVENT, that, false);
621         document.removeEventListener('webkitTransitionEnd', that, false);
622 
623         if (that.options.checkDOMChanges) {
624             that.element.removeEventListener('DOMSubtreeModified', that, false);
625         }
626 
627         if (that.scrollBarX) {
628             that.scrollBarX = that.scrollBarX.remove();
629         }
630 
631         if (that.scrollBarY) {
632             that.scrollBarY = that.scrollBarY.remove();
633         }
634         
635         if (full) {
636             that.wrapper.parentNode.removeChild(that.wrapper);
637         }
638         
639         return null;
640     }
641 };

645 function scrollbar (dir, wrapper, fade, shrink, color) {
646     var that = this,
647         doc = document;
648     
649     that.dir = dir;
650     that.fade = fade;
651     that.shrink = shrink;
652     that.uid = ++uid;
653 
654     // Create main scrollbar
655     that.bar = doc.createElement('div');
656 
657     that.bar.style.cssText = 'position:absolute;top:0;left:0;-webkit-transition-timing-function:cubic-bezier(0,0,0.25,1);pointer-events:none;-webkit-transition-duration:0;-webkit-transition-delay:0;-webkit-transition-property:-webkit-transform;z-index:10;background:' + color + ';' +
658         '-webkit-transform:' + translateOpen + '0,0' + translateClose + ';' +
659         (dir == 'horizontal' ? '-webkit-border-radius:3px 2px;min-width:6px;min-height:5px' : '-webkit-border-radius:2px 3px;min-width:5px;min-height:6px');
660 
661     // Create scrollbar wrapper
662     that.wrapper = doc.createElement('div');
663     that.wrapper.style.cssText = '-webkit-mask:-webkit-canvas(scrollbar' + that.uid + that.dir + ');position:absolute;z-index:10;pointer-events:none;overflow:hidden;opacity:0;-webkit-transition-duration:' + (fade ? '300ms' : '0') + ';-webkit-transition-delay:0;-webkit-transition-property:opacity;' +
664         (that.dir == 'horizontal' ? 'bottom:2px;left:2px;right:7px;height:5px' : 'top:2px;right:2px;bottom:7px;width:5px;');
665 
666     // Add scrollbar to the DOM
667     that.wrapper.appendChild(that.bar);
668     wrapper.appendChild(that.wrapper);
669 }

674 /**

677 {
678     init: function (scroll, size) {
679         var that = this,
680             doc = document,
681             pi = Math.PI,
682             ctx;
683 
684         // Create scrollbar mask
685         if (that.dir == 'horizontal') {
686             if (that.maxSize != that.wrapper.offsetWidth) {
687                 that.maxSize = that.wrapper.offsetWidth;
688                 ctx = doc.getCSSCanvasContext("2d", "scrollbar" + that.uid + that.dir, that.maxSize, 5);
689                 ctx.fillStyle = "rgb(0,0,0)";
690                 ctx.beginPath();
691                 ctx.arc(2.5, 2.5, 2.5, pi/2, -pi/2, false);
692                 ctx.lineTo(that.maxSize-2.5, 0);
693                 ctx.arc(that.maxSize-2.5, 2.5, 2.5, -pi/2, pi/2, false);
694                 ctx.closePath();
695                 ctx.fill();
696             }
697         } else {
698             if (that.maxSize != that.wrapper.offsetHeight) {
699                 that.maxSize = that.wrapper.offsetHeight;
700                 ctx = doc.getCSSCanvasContext("2d", "scrollbar" + that.uid + that.dir, 5, that.maxSize);
701                 ctx.fillStyle = "rgb(0,0,0)";
702                 ctx.beginPath();
703                 ctx.arc(2.5, 2.5, 2.5, pi, 0, false);
704                 ctx.lineTo(5, that.maxSize-2.5);
705                 ctx.arc(2.5, that.maxSize-2.5, 2.5, 0, pi, false);
706                 ctx.closePath();
707                 ctx.fill();
708             }
709         }
710 
711         that.size = Math.max(Math.round(that.maxSize * that.maxSize / size), 6);
712         that.maxScroll = that.maxSize - that.size;
713         that.toWrapperProp = that.maxScroll / (scroll - size);
714         that.bar.style[that.dir == 'horizontal' ? 'width' : 'height'] = that.size + 'px';
715     },
716     
717     setPosition: function (pos) {
718         var that = this;
719         
720         if (that.wrapper.style.opacity != '1') {
721             that.show();
722         }
723 
724         pos = Math.round(that.toWrapperProp * pos);
725 
726         if (pos < 0) {
727             pos = that.shrink ? pos + pos*3 : 0;
728             if (that.size + pos < 7) {
729                 pos = -that.size + 6;
730             }
731         } else if (pos > that.maxScroll) {
732             pos = that.shrink ? pos + (pos-that.maxScroll)*3 : that.maxScroll;
733             if (that.size + that.maxScroll - pos < 7) {
734                 pos = that.size + that.maxScroll - 6;
735             }
736         }
737 
738         pos = that.dir == 'horizontal'
739             ? translateOpen + pos + 'px,0' + translateClose
740             : translateOpen + '0,' + pos + 'px' + translateClose;
741         that.bar.style.webkitTransform = pos;
742     },
743     
744     
745 
746     show: function () {
747         if (has3d) {
748             this.wrapper.style.webkitTransitionDelay = '0';
749         }
750         this.wrapper.style.opacity = '1';
751     },
752 
753     hide: function () {
754         if (has3d) {
755             this.wrapper.style.webkitTransitionDelay = '350ms';
756         }
757         this.wrapper.style.opacity = '0';
758     },
759     
760     remove: function () {
761         this.wrapper.parentNode.removeChild(this.wrapper);
762         return null;
763     }
