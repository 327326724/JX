  9 Jx().$package(function (J) {
 10     var $D = J.dom,
 11         $E = J.event;
 12     J.ui = J.ui || {};
 13     /**
 14      * 【Boxy】
 15      * 
 16      * @class 
 17      * @memberOf ui
 18      * @name Boxy
 19      * @constructor
 20      * @param {Object} option 参数对象
 21      */
 22     var Boxy = new J.Class(
 23     /**
 24      * @lends ui.Boxy.prototype
 25      */ 
 26     {
 27         /**
 28          * @ignore
 29          */
 30         init: function (option) {
 31             var _this = this;
 32             option = option || {};
 33             this._id = (option.id || (new Date().getTime()));
 34             option.name = option.id;
 35             option.width = option.width || 300;
 36             option.height = option.height || 300;
 37             option.appendTo = option.appendTo || document.body;
 38             option.zIndex = option.zIndex || 1;
 39             var node = $D.node('div',{
 40                 'class':'ui_boxy'
 41             });
 42             node.innerHTML = '\
 43                 <div style="position:relative; z-index:1;"><div class="ui_boxyClose" id="ui_boxyClose_'+this._id+'"></div></div>\
 44                 <div class="ui_boxyWrapper" id="ui_boxyWrapper_'+this._id+'"></div>\
 45                 ';
 46             option.appendTo.appendChild(node);
 47             option.container = node;
 48             option.body = $D.id('ui_boxyWrapper_'+this._id);
 49             this._option = option;
 50             
 51             this._panel = new J.ui.Panel(option);
 52             this._panel.setWidth(option.width);
 53             this._panel.setHeight(option.height);
 54             this._maskLayer;
 55             if(option.modal){
 56                 this._maskLayer = new J.ui.MaskLayer({ 
 57                         appendTo: option.appendTo,
 58                         zIndex: option.zIndex,
 59                         opacity: 0.5
 60                     });
 61                 this._maskLayer.show();
 62             }
 63             this._panel.setZIndex(option.zIndex + 1);
 64             this.setCenter(option);
 65             new J.ui.Drag(node, node, {isLimited: true});
 66             
 67             var closeBtn = $D.id("ui_boxyClose_" + this._id);
 68             var wrapper = this._wrapper = $D.id("ui_boxyWrapper_" + this._id);
 69             $E.on(closeBtn, 'click', function(e){
 70                 //_this.hide();
 71                 _this.close();
 72                 option.onClose && option.onClose.apply(_this);
 73                 $E.notifyObservers(_this, 'close'); 
 74             });
 75             
 76         },
 77         /**
 78          * 获取弹出框所处的面板
 79          * @return {ui.Panel} 
 80          */
 81         getPanel: function(){
 82             return this._panel;
 83         },
 84         /**
 85          * 显示弹出框
 86          */
 87         show: function(){
 88             this._panel.show(); 
 89         },
 90         /**
 91          * 隐藏弹出框
 92          */
 93         hide: function(){
 94             this._panel.hide();
 95         },
 96         /**
 97          * 设置弹出框的层级
 98          * @param {Number} index 层级
 99          */
100         setZIndex: function(index){
101             this._panel.setZIndex(index); 
102         },
103         /**
104          * 将弹出框居中
105          */
106         setCenter: function(option){
107             var w = $D.getClientWidth(),
108                 h = $D.getClientHeight();
109             var l = (w > option.width) ? (w - option.width) / 2 : 0;
110             var t = (h > option.height) ? (h - option.height) / 2 : 0;
111             this._panel.setXY(l,t);
112         },
113         /**
114          * 指示弹出框是否显示
115          */
116         isShow: function(){
117             return this._panel.isShow();
118         },
119         /**
120          * 关闭弹出框
121          */
122         close: function(){
123             //this._panel.hide();
124             this._maskLayer && this._maskLayer.remove();
125             this._option.appendTo.removeChild(this._option.container);
126         }
127     });
128     
129     J.ui.Boxy = Boxy;
130 });
