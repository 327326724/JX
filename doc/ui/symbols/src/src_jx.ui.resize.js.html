1 /**
  2  * Resize 模块
  3  */
  4 Jx().$package(function (J) {
  5     J.ui = J.ui || {};
  6     var $D = J.dom,
  7         $E = J.event;
  8 
  9     var id = 0;
 10     var handleNames = {
 11         t: "t",
 12         r: "r",
 13         b: "b",
 14         l: "l",
 15         rt: "rt",
 16         rb: "rb",
 17         lb: "lb",
 18         lt: "lt"
 19     };
 20         
 21     var clientHeight = 0;
 22     var clientWidth = 0;
 23     /**
 24     * resize类
 25     * 
 26     * @memberOf ui
 27     * @class
 28     * @name Resize
 29     * @constructor
 30     * @param {Element} apperceiveEl 监听resize动作的元素
 31     * @param {Element} effectEl 展现resize结果的元素
 32     * @param {Object} option 其他选项，如:dragProxy,size,minWidth...
 33     * 
 34     */
 35     J.ui.Resize = new J.Class(
 36     /**
 37      * @lends ui.Resize.prototype
 38      */
 39     {
 40         /**
 41          * @ignore
 42          */
 43         init: function (apperceiveEl, effectEl, option) {
 44             var context = this;
 45             option = option || {};
 46 
 47             this.apperceiveEl = apperceiveEl;
 48             if (effectEl === false) {
 49                 this.effectEl = false;
 50             } else {
 51                 this.effectEl = effectEl || apperceiveEl;
 52             }
 53             
 54             this.size = option.size || 5;
 55             this.minWidth = option.minWidth || 0;
 56             this.minHeight = option.minHeight || 0;
 57             this._dragProxy = option.dragProxy;
 58             this.isLimited = option.isLimited || false;
 59             if (this.isLimited) {
 60                 this._leftMargin = option.leftMargin || 0;
 61                 this._topMargin = option.topMargin || 0;
 62                 this._rightMargin = option.rightMargin || 0;
 63                 this._bottomMargin = option.bottomMargin || 0;
 64             }
 65 
 66             this._left = this.getLeft();
 67             this._top = this.getTop();
 68             this._width = this.getWidth();
 69             this._height = this.getHeight();
 70 
 71             this.id = this.getId();
 72 
 73             var styles = {
 74                 t: "cursor:n-resize; z-index:1; left:0; top:-5px; width:100%; height:5px;",
 75                 r: "cursor:e-resize; z-index:1; right:-5px; top:0; width:5px; height:100%;",
 76                 b: "cursor:s-resize; z-index:1; left:0; bottom:-5px; width:100%; height:5px;",
 77                 l: "cursor:w-resize; z-index:1; left:-5px; top:0; width:5px; height:100%;",
 78                 rt: "cursor:ne-resize; z-index:2; right:-5px; top:-5px; width:5px; height:5px;",
 79                 rb: "cursor:se-resize; z-index:2; right:-5px; bottom:-5px; width:5px; height:5px;",
 80                 lt: "cursor:nw-resize; z-index:2; left:-5px; top:-5px; width:5px; height:5px;",
 81                 lb: "cursor:sw-resize; z-index:2; left:-5px; bottom:-5px; width:5px; height:5px;"
 82             };
 83 
 84             this._onMousedown = function () {
 85                 $E.notifyObservers(context, "mousedown", { width: context._width, height: context._height });
 86             };
 87             this._onDragEnd = function () {
 88 //                J.out("this._width： " + context._width);
 89 //              J.out("this._height： " + context._height);
 90                 $E.notifyObservers(context, "end", {
 91                     x: context.getLeft(),
 92                     y: context.getTop(),
 93                     width: context.getWidth(),
 94                     height: context.getHeight()
 95                 });
 96             };
 97 
 98             for (var p in handleNames) {
 99                 var tempEl = $D.node("div", {
100                     "id": "window_" + this.id + "_resize_" + handleNames[p]
101                 });
102 
103                 this.apperceiveEl.appendChild(tempEl);
104                 $D.setCssText(tempEl, "position:absolute; overflow:hidden; background:url(" + J.path + "style/image/transparent.gif);" + styles[p]);
105                 if (this._dragProxy) {
106                     //$E.on(tempEl, "mousedown", this._onMousedown);
107                 } else {
108 
109                 }
110 
111                 this["_dragController_" + handleNames[p]] = new J.ui.Drag(tempEl, false);
112 
113             }
114 
115 
116 
117             // 左侧
118             this._onDragLeftStart = function (xy) {
119                 $E.notifyObservers(context, "mousedown", { width: context._width, height: context._height });
120                 context._startLeft = context._left = context.getLeft();
121                 context._startWidth = context._width = context.getWidth();
122                 context._startHeight = context._height = context.getHeight();
123             };
124             this._onDragLeft = function (xy) {
125                 var w = context._startWidth - xy.x;
126                 var x = context._startLeft + xy.x;
127                 if (w < context.minWidth) {
128                     w = context.minWidth;
129                     x = context._startLeft + (context._startWidth - w);
130                 }
131                 if (context.isLimited && (x - context._leftMargin) < 0) {
132                     x = context._leftMargin;
133                     w = context._startLeft + context._startWidth - context._leftMargin;
134                 }
135                 context.setLeft(x);
136                 context.setWidth(w);
137                 $E.notifyObservers(context, "resize", { width: context._width, height: context._height });
138 
139             };
140 
141             // 上侧
142             this._onDragTopStart = function (xy) {
143                 $E.notifyObservers(context, "mousedown", { width: context._width, height: context._height });
144                 context._startTop = context._top = context.getTop();
145                 context._startHeight = context._height = context.getHeight();
146             };
147             this._onDragTop = function (xy) {
148                 var h = context._startHeight - xy.y;
149                 var y = context._startTop + xy.y;
150                 if (h < context.minHeight) {
151                     h = context.minHeight;
152                     y = context._startTop + (context._startHeight - h);
153                 }
154                 if (context.isLimited && (y - context._topMargin) < 0) {
155                     y = context._topMargin;
156                     h = context._startTop + context._startHeight - context._topMargin;
157                 }
158                 context.setTop(y);
159                 context.setHeight(h);
160                 $E.notifyObservers(context, "resize", { width: context._width, height: context._height });
161             };
162 
163 
164 
165             // 右侧
166             this._onDragRightStart = function (xy) {
167                  $E.notifyObservers(context, "mousedown", { width: context._width, height: context._height });
168                 context._startWidth = context._width = context.getWidth();
169                 context._startLeft = context._left = context.getLeft();
170                 context._startHeight = context._height = context.getHeight();
171                 clientWidth = qqweb.layout.getClientWidth();
172             };
173             this._onDragRight = function (xy) {
174                 var w = context._startWidth + xy.x;
175                 if (w < context.minWidth) {
176                     w = context.minWidth;
177                 }
178                 var clientWidth = $D.getClientWidth() || 0;
179                 var maxWidth = clientWidth - context._startLeft - context._rightMargin;
180                 if (context.isLimited && maxWidth < w) {
181                     w = maxWidth;
182                 }
183                 context.setWidth(w);
184                 $E.notifyObservers(context, "resize", { width: context._width, height: context._height });
185             };
186 
187 
188             // 下侧
189             this._onDragBottomStart = function (xy) {
190                 $E.notifyObservers(context, "mousedown", { width: context._width, height: context._height });
191                 context._startHeight = context._height = context.getHeight();
192                 context._startTop = context._top = context.getTop();
193                 clientHeight = $D.getClientHeight();
194             };
195             this._onDragBottom = function (xy) {
196                 var h = context._startHeight + xy.y;
197                 if (h < context.minHeight) {
198                     h = context.minHeight;
199                 }
200                 var clientHeight = $D.getClientHeight() || 0;
201                 var maxHeight = clientHeight - context._startTop - context._bottomMargin;
202                 if (context.isLimited && maxHeight < h) {
203                     h = maxHeight;
204                 }
205                 context.setHeight(h);
206                 $E.notifyObservers(context, "resize", { width: context._width, height: context._height });
207             };
208 
209             // 左上
210             this._onDragLeftTopStart = function (xy) {
211                 context._onDragLeftStart(xy);
212                 context._onDragTopStart(xy);
213             };
214             this._onDragLeftTop = function (xy) {
215                 context._onDragLeft(xy);
216                 context._onDragTop(xy);
217             };
218 
219             // 左下
220             this._onDragLeftBottomStart = function (xy) {
221                 context._onDragLeftStart(xy);
222                 context._onDragBottomStart(xy);
223             };
224             this._onDragLeftBottom = function (xy) {
225                 context._onDragLeft(xy);
226                 context._onDragBottom(xy);
227             };
228 
229 
230             // 右下
231             this._onDragRightBottomStart = function (xy) {
232                 context._onDragRightStart(xy);
233                 context._onDragBottomStart(xy);
234             };
235             this._onDragRightBottom = function (xy) {
236                 context._onDragRight(xy);
237                 context._onDragBottom(xy);
238             };
239 
240             // 右上
241             this._onDragRightTopStart = function (xy) {
242                 context._onDragRightStart(xy);
243                 context._onDragTopStart(xy);
244             };
245             this._onDragRightTop = function (xy) {
246                 context._onDragRight(xy);
247                 context._onDragTop(xy);
248             };
249 
250 
251 
252             $E.addObserver(this["_dragController_" + handleNames.t], "start", this._onDragTopStart);
253             $E.addObserver(this["_dragController_" + handleNames.t], "move", this._onDragTop);
254             $E.addObserver(this["_dragController_" + handleNames.t], "end", this._onDragEnd);
255 
256             $E.addObserver(this["_dragController_" + handleNames.r], "start", this._onDragRightStart);
257             $E.addObserver(this["_dragController_" + handleNames.r], "move", this._onDragRight);
258             $E.addObserver(this["_dragController_" + handleNames.r], "end", this._onDragEnd);
259 
260             $E.addObserver(this["_dragController_" + handleNames.b], "start", this._onDragBottomStart);
261             $E.addObserver(this["_dragController_" + handleNames.b], "move", this._onDragBottom);
262             $E.addObserver(this["_dragController_" + handleNames.b], "end", this._onDragEnd);
263 
264             $E.addObserver(this["_dragController_" + handleNames.l], "start", this._onDragLeftStart);
265             $E.addObserver(this["_dragController_" + handleNames.l], "move", this._onDragLeft);
266             $E.addObserver(this["_dragController_" + handleNames.l], "end", this._onDragEnd);
267 
268 
269 
270             $E.addObserver(this["_dragController_" + handleNames.rb], "start", this._onDragRightBottomStart);
271             $E.addObserver(this["_dragController_" + handleNames.rb], "move", this._onDragRightBottom);
272             $E.addObserver(this["_dragController_" + handleNames.rb], "end", this._onDragEnd);
273 
274             $E.addObserver(this["_dragController_" + handleNames.rt], "start", this._onDragRightTopStart);
275             $E.addObserver(this["_dragController_" + handleNames.rt], "move", this._onDragRightTop);
276             $E.addObserver(this["_dragController_" + handleNames.rt], "end", this._onDragEnd);
277 
278             $E.addObserver(this["_dragController_" + handleNames.lt], "start", this._onDragLeftTopStart);
279             $E.addObserver(this["_dragController_" + handleNames.lt], "move", this._onDragLeftTop);
280             $E.addObserver(this["_dragController_" + handleNames.lt], "end", this._onDragEnd);
281 
282             $E.addObserver(this["_dragController_" + handleNames.lb], "start", this._onDragLeftBottomStart);
283             $E.addObserver(this["_dragController_" + handleNames.lb], "move", this._onDragLeftBottom);
284             $E.addObserver(this["_dragController_" + handleNames.lb], "end", this._onDragEnd);
285         },
286         /**
287          * 设置effectEl的宽度
288          * @param {Number} w
289          */
290         setWidth: function (w) {
291             $D.setStyle(this.effectEl, "width", w + "px");
292             this._width = w;
293         },
294         /**
295          * 设置effectEl的高度
296          * @param {Number} h
297          */
298         setHeight: function (h) {
299             $D.setStyle(this.effectEl, "height", h + "px");
300             this._height = h;
301         },
302         /**
303          * 设置effectEl的x坐标
304          * @param {Number} x
305          */
306         setLeft: function (x) {
307             $D.setStyle(this.effectEl, "left", x + "px");
308             this._left = x;
309         },
310         /**
311          * 设置effectEl的y坐标
312          * @param {Number} y
313          */
314         setTop: function (y) {
315             $D.setStyle(this.effectEl, "top", y + "px");
316             this._top = y;
317         },
318         /**
319          * 获取effectEl的宽度
320          * @return {Number}
321          */
322         getWidth: function () {
323             return parseInt($D.getStyle(this.effectEl, "width"));
324         },
325         /**
326          * 获取effectEl的高度
327          * @return {Number}
328          */
329         getHeight: function () {
330             return parseInt($D.getStyle(this.effectEl, "height"));
331         },
332         /**
333          * 获取effectEl的x
334          * @return {Number}
335          */
336         getLeft: function () {
337             return parseInt($D.getStyle(this.effectEl, "left"));
338         },
339         /**
340          * 获取effectEl的y
341          * @return {Number}
342          */
343         getTop: function () {
344             return parseInt($D.getStyle(this.effectEl, "top"));
345         },
346         /**
347          * @private
348          */
349         getId: function () {
350             return id++;
351         },
352         /**
353          * 锁定resize
354          */
355         lock: function () {
356             for (var p in handleNames) {
357                 this["_dragController_" + handleNames[p]].lock();
358             }
359         },
360         /**
361          * 解锁resize
362          */
363         unlock: function () {
364             for (var p in handleNames) {
365                 this["_dragController_" + handleNames[p]].unlock();
366             }
367         },
368         /**
369          * 显示触发resize的节点
370          */
371         show: function () {
372             for (var p in handleNames) {
373                 this["_dragController_" + handleNames[p]].show();
374             }
375         },
376         /**
377          * 隐藏触发resize的节点
378          */
379         hide: function () {
380             for (var p in handleNames) {
381                 this["_dragController_" + handleNames[p]].hide();
382             }
383         },
384         /**
385          * 设置resize的限定边界
386          */
387         setLimite : function(option){
388             option = option || {};
389             this.isLimited = option.isLimited || true;
390             if (this.isLimited) {
391                 this._leftMargin = option.leftMargin || 0;
392                 this._topMargin = option.topMargin || 0;
393                 this._rightMargin = option.rightMargin || 0;
394                 this._bottomMargin = option.bottomMargin || 0;
395             }
396         },
397         /**
398          * 设置最小宽高
399          */
400         setMinLimite : function(option){
401             option = option||{};
402             this.minWidth = option.minWidth||0;
403             this.minHeight = option.minHeight||0;
404         }
405     });
