1 /**
  2  * MaskLayer模块
  3 **/
  4 Jx().$package(function(J){
  5     var $ = J.dom.id,
  6         $D = J.dom,
  7         $E = J.event;
  8         
  9     J.ui = J.ui || {};
 10     
 11     var _id = 0;
 12     /**
 13      * MaskLayer 遮罩层类
 14      * 
 15      * @memberOf ui
 16      * @class
 17      * @name MaskLayer
 18      * @constructor
 19      * @param {Object} option 其他选项，如:zIndex,appendTo...
 20      * 
 21      **/
 22     J.ui.MaskLayer = new J.Class(
 23     /**
 24      * @lends ui.MaskLayer.prototype
 25      */
 26     {
 27         _getMaskId: function(){
 28             return _id++;   
 29         },
 30         /**
 31          * @ignore
 32          */
 33         init:function(option){
 34             option = option || {};
 35             var mid = this._getMaskId();
 36             var context = this;
 37             this._initZIndex = option.zIndex = !J.isUndefined(option.zIndex) ? option.zIndex : 9000000;
 38             this._initOpacity = option.opacity = !J.isUndefined(option.opacity) ? option.opacity : 0.5;
 39             option.appendTo = option.appendTo || document.body;
 40             option.className = option.className || '';
 41             this.option = option;
 42             this.container = $D.node("div", {
 43                 "id": "ui_maskLayer_" + mid,
 44                 "class" : "ui_maskLayer " + option.className
 45             });
 46             var html = '<div id="ui_maskLayerBody_' + mid + '" class="ui_maskLayerBody"></div>';
 47             if(J.browser.ie){
 48                 html += '<iframe class="ui_maskBgIframe"></iframe>';
 49             }
 50             this.container.innerHTML = html;
 51             
 52             this.reset();
 53             option.appendTo.appendChild(this.container);
 54             
 55             var observer = {
 56                 /**
 57                  * @ignore
 58                  */
 59                 onMaskLayerClick : function(){
 60                     $E.notifyObservers(context, "click", context);
 61                 }
 62             };
 63             
 64             $E.on(this.container, "click", observer.onMaskLayerClick);
 65             
 66             this.body = $D.id("ui_maskLayerBody_" + mid);
 67         },
 68         /**
 69          * 获取遮罩的dom节点
 70          * @return {HTMLElement} 
 71          */
 72         getElement : function(){
 73             return this.container;  
 74         },
 75         /**
 76          * 添加一个dom到遮罩中
 77          */
 78         append : function(el){
 79             this.body.appendChild(el);
 80         },
 81         /**
 82          * 显示
 83          */
 84         show : function(){
 85             $D.show(this.container);
 86             $E.notifyObservers(this, "show");
 87             this._isShow = true;
 88         },
 89         /**
 90          * 隐藏
 91          */
 92         hide : function(){
 93             $D.hide(this.container);
 94             $E.notifyObservers(this, "hide");
 95             this._isShow = false;
 96         },
 97         /**
 98          * 指示遮罩是否显示中
 99          */
100         isShow : function(){
101             return this._isShow;
102         },
103         /**
104          * 切换遮罩的显示隐藏
105          */
106         toggleShow : function(){
107             if(this.isShow()){
108                 this.hide();
109             }else{
110                 this.show();
111             }
112         },
113         /**
114          * 获取遮罩的层级
115          */
116         getZIndex : function(){
117             return this._zIndex;
118         },
119         /**
120          * 设置遮罩的层级
121          */
122         setZIndex : function(zIndex){
123             $D.setStyle(this.container, "zIndex", zIndex);
124             this._zIndex = zIndex;
125         },
126         /**
127          * 设置遮罩的透明度
128          * @param {Decimal} opacity 透明度,0~1之间的小数
129          */
130         setOpacity: function(opacity){
131             $D.setStyle(this.container, 'opacity', opacity);
132         },
133         /**
134          * 重置遮罩的透明度,zindex等,重新设置为初始值
135          * @return {MaskLayer} 返回自身
136          */
137         reset: function(){
138             this.setOpacity(this._initOpacity);
139             this.setZIndex(this._initZIndex);
140             return this;
141         },
142         /**
143          * 淡入显示遮罩
144          */
145         fadeIn : function(){
146             this.show();
147         },
148         /**
149          * 淡出显示遮罩
150          */
151         fadeOut : function(){
152             this.hide();
153         },
154         /**
155          * 移除遮罩
156          */
157         remove : function(){
158             if(this.option.appendTo)    {
159                 this.option.appendTo.removeChild(this.container);
160             }
161         }
162     });
163 
164 });
