 12 Jx().$package(function (J) {
 13     var $ = J.dom.id,
 14         $D = J.dom,
 15         $S = J.string,
 16         $A = J.array,
 17         $E = J.event,
 18         $T = J.fx.transitions;
 19         
 20     // Tooltip类
 21     /**
 22      * 【Tooltip】
 23      * 
 24      * @class 
 25      * @memberOf ui
 26      * @constructor
 27      * @name Tooltip
 28      * @param {Element} els 需要生成tooltip的DOM节点
 29      * @param {Object} option 参数对象 option:{text:'tooltip内容，默认去title属性'}
 30      * @since version 1.0
 31      * @description Jx.ui.Tooltip是一个长方形的小提示，在用户将指针悬停在一个控件上时显示有关该控件用途的简短说明。
 32      */
 33     var Tooltip = new J.Class(
 34     /**
 35      * @lends ui.Tooltip.prototype
 36      */ 
 37     {
 38         /**
 39         * @ignore
 40         */
 41         init: function (els, option) {
 42             var tooltipContext = this;
 43             option = option || {};
 44             
 45             tooltipContext.id = Tooltip.count;
 46             Tooltip.count++;
 47             
 48             $A.forEach(els, function(el,i,els){
 49                 var text = option.text || el.getAttribute("title");
 50                 
 51                 // 检测文字所需要的宽度
 52                 var width;
 53                 var tempEl = $D.node("div",{
 54                     style:"float:left;width:auto;background:blue;visibility:hidden;"
 55                 });
 56                 document.body.appendChild(tempEl);
 57                 tempEl.innerHTML = text;
 58                 width = $D.getClientWidth(tempEl, "width");
 59                 width += (width % 2);
 60                 document.body.removeChild(tempEl);
 61                 tempEl = null;
 62                 
 63                 
 64                 
 65                 el.removeAttribute("title");
 66 
 67                 if(!J.isUndefined(text)) {
 68                     /**
 69                      * @ignore
 70                      */
 71                     tooltipContext.mouseMove = function(e){
 72                         var tooltipEL = tooltipContext.tooltipEL;
 73                         var tipX = e.pageX + 24;
 74                         var tipY = e.pageY - 18;
 75                         var tipWidth = $D.getWidth(tooltipEL);
 76                         var tipHeight = $D.getHeight(tooltipEL);
 77 //                      if(tipX + tipWidth > $D.getScrollLeft(document.body) + $D.getWidth(document.body)){
 78 //                          tipX = e.pageX - tipWidth;
 79 //                      }
 80 //                      if($D.getHeight(document.body)+$D.getScrollTop(document.body) < tipY + tipHeight){
 81 //                          tipY = e.pageY - tipHeight;
 82 //                      }
 83                         $D.setXY(tooltipEL, tipX, tipY);
 84                     };
 85                     
 86                     $E.on(el, "mouseover", function(e){
 87                         var tipX = e.pageX + 24;
 88                         var tipY = e.pageY - 18;
 89                         if(!tooltipContext.tooltipEL){
 90                             tooltipContext.createTooltip();
 91                         }
 92                         var tooltipTextEL = tooltipContext.tooltipTextEL;
 93                         tooltipTextEL.innerHTML = text;
 94                         
 95                         tooltipContext.showIn({width:width});
 96                         
 97                     });
 98                     
 99                     $E.on(el, "mouseout", function(e){
100                         if(tooltipContext.tooltipEL){
101                             tooltipContext.showOut({width:width});
102                         }
103                     });
104                 }
105             });
106             
107         },
108         
109         /**
110         * @private
111         */
112         createTooltip: function(){
113             var tooltipContext = this;
114             var tooltipEL;
115             var tooltipTextEL;
116             var tempLeftEL;
117             var tempRightEL;
118             
119             
120             var tooltipTextELBg = 'background:url(http://qplus1.idqqimg.com/jx/assets/images/tooltip_c.png?t=20111011001);',
121                 tooltipLeftELBg =  'background:url(http://qplus1.idqqimg.com/jx/assets/images/tooltip.png?t=20111011001) no-repeat;',
122                 tooltipRightELBg = ' background:url(http://qplus1.idqqimg.com/jx/assets/images/tooltip.png?t=20111011001) no-repeat right';
123             
124             if(J.browser.ie && J.browser.ie == 6){
125                 tooltipTextELBg = 'filter: progid:DXImageTransform.Microsoft.AlphaImageLoader(src="http://qplus1.idqqimg.com/jx/assets/images/tooltip_c.png?t=20111011001", sizingMethod="scale");';
126                 tooltipLeftELBg = 'filter: progid:DXImageTransform.Microsoft.AlphaImageLoader(src="http://qplus1.idqqimg.com/jx/assets/images/tooltip_l.png?t=20111011001", sizingMethod="scale");';
127                 tooltipRightELBg = 'filter: progid:DXImageTransform.Microsoft.AlphaImageLoader(src="http://qplus1.idqqimg.com/jx/assets/images/tooltip_r.png?t=20111011001", sizingMethod="scale");';
128             }
129             
130             tooltipContext.tooltipEL = tooltipEL = $D.node("div",{
131                 id:"jxTooltip_"+tooltipContext.id,
132                 style:'position: absolute; margin:10px; width:107px; height:38px; z-index: 10000; display: none;'
133             });
134             
135             tooltipContext.tooltipTextEL = tooltipTextEL = $D.node("div",{
136                 id:"jxTooltipText_"+tooltipContext.id,
137                 style:'line-height:30px; width:100%; height:38px; color:white; overflow:hidden;' + tooltipTextELBg
138             });
139 
140             tempLeftEL = $D.node("div",{
141                 style:'position:absolute; left:-12px; width:12px; height:38px; overflow:hidden;' + tooltipLeftELBg 
142             });
143             tempRightEL = $D.node("div",{
144                 style:'position:absolute; right:-12px; width:12px; height:38px; overflow:hidden;' + tooltipRightELBg
145             });
146             
147             tooltipEL.appendChild(tempLeftEL);
148             tooltipEL.appendChild(tempRightEL);
149             tooltipEL.appendChild(tooltipTextEL);
150 
151             document.body.appendChild(tooltipEL);
152             //$D.setStyle(tooltipEL, "opacity", 0);
153             tooltipContext.showInAnimation = new J.fx.Animation({
154                 element:tooltipEL,
155                 property:"width",
156                 from:0,
157                 to:1,
158                 unit:"px",
159                 duration:300,
160                 transition:$T.sinusoidal.easeOut
161             });
162             
163             tooltipContext.showOutAnimation = new J.fx.Animation({
164                 element:tooltipEL,
165                 property:"width",
166                 from:1,
167                 to:0,
168                 unit:"px",
169                 duration:300,
170                 transition:$T.sinusoidal.easeOut
171             });
172             
173             $E.addObserver(tooltipContext.showOutAnimation,"end", function(){
174                 $D.hide(tooltipEL);
175             });
176             
177         },
178         
179         /**
180         * 显示tooltip
181         * @param {Object} option 参数对象 option:{width:'tooltip宽度'}
182         * @return 
183         */
184         showIn:function(option){
185             var tooltipContext = this;
186             var tooltipEL = tooltipContext.tooltipEL;
187             
188             tooltipContext.showOutAnimation.end();
189             $E.on(document, "mousemove", tooltipContext.mouseMove);
190             $D.show(tooltipEL);
191             tooltipContext.showInAnimation.start(0,option.width);
192         },
193         
194         /**
195         * 隐藏tooltip
196         * @param {Object} option 参数对象 option:{width:'tooltip宽度'}
197         * @return 
198         */
199         showOut:function(option){
200             var tooltipContext = this;
201             var tooltipEL = tooltipContext.tooltipEL;
202             
203             tooltipContext.showInAnimation.end();
204             //alert(option.width)
205 
206             tooltipContext.showOutAnimation.start(option.width,0);
207             $E.addObserver(tooltipContext.showOutAnimation, 'end', function(e){
208                 $E.off(document, "mousemove", tooltipContext.mouseMove);
209             });
210             //$D.hide(tooltipEL);
211         },
212         
213         /**
214         * 关于tooltip
215         * @private
216         */
217         about:function(){
218             alert("Jx().ui.Tooltip() by Jetyu of Tencent!");
219         }
220        
221 
222     });

235     J.ui = J.ui || {};
236     J.ui.Tooltip = Tooltip;
