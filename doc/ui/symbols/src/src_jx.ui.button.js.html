  8  Jx().$package(function (J) {
  9     var $D = J.dom,
 10         $S = J.string,
 11         $E = J.event;
 12     var _id = 0;
 13     J.ui = J.ui || {};
 14     /**
 15      * 【Button】
 16      * 
 17      * @class 
 18      * @memberOf ui
 19      * @name Button
 20      * @constructor
 21      * @param {Object} option 参数对象
 22      */
 23     J.ui.Button = new J.Class(
 24     /**
 25      * @lends ui.Button.prototype
 26      */
 27     {
 28         _class: 'ui_button',
 29         _available: true, // available for user?
 30         _shownInLogic: false, // should be shown in logic
 31         _getButtionId: function(){
 32             return _id++;   
 33         },
 34         /**
 35          * @ignore
 36          */
 37         init: function (option){
 38             option = option || {};
 39             var context = this;
 40             var bid = this._getButtionId();
 41             var buttonOption = {
 42                 appendTo: $D.getDocumentElement(),
 43                 className: '',
 44                 text: '',
 45                 title : ''
 46             };
 47             J.extend(buttonOption, option);
 48             
 49             this._node = $D.node('a', {
 50                  'id': 'ui_button_' + bid,
 51                  'class': this._class + ' ' + $S.encodeScript(buttonOption.className),
 52                  'title': $S.encodeScript(buttonOption.title),
 53                  'hidefocus': '',
 54                  'href': '###'
 55             });
 56             this._node.innerHTML = $S.encodeHtml(buttonOption.text);
 57             buttonOption.appendTo.appendChild(this._node);
 58             if(buttonOption.event){
 59                 this.attachEvent(buttonOption.event);
 60             }
 61             if(buttonOption.isStopPropagation){
 62                 $E.on(this._node,"mousedown",function(e){
 63                     e.stopPropagation();
 64                 })
 65                 $E.on(this._node,"click",function(e){
 66                     e.stopPropagation();
 67                 })
 68                 
 69             }
 70         },
 71         /**
 72          * 自定义事件绑定
 73          * @param eventObj {Object} 事件对象
 74          */
 75         attachEvent : function(eventObj){
 76             for(var i in eventObj){
 77                 $E.on(this._node, i, eventObj[i]);
 78             }
 79         },
 80         /**
 81          * 移除自定义事件
 82          * @param eventObj {Object} 事件对象
 83          */
 84         removeEvent : function(eventObj){
 85             for(var i in eventObj){
 86                 $E.off(this._node, i, eventObj[i]);
 87             }
 88         },
 89         /**
 90          * 设置按钮是否可用
 91          * @param {Boolean} val
 92          */
 93         setAvailability: function(val){
 94             // change the availability
 95             this._available = !!val;
 96 
 97             // show/hide the button by logic
 98             this._shownInLogic && $D[this._available ? 'show' : 'hide'](this._node);
 99         },
100         /**
101          * 隐藏按钮
102          */
103         hide: function(){
104             // shown in logic?
105             this._shownInLogic = false;
106 
107             // do nothing if it's unavailable
108             if (!this._available) {
109                 return;
110             }
111 
112             $D.hide(this._node);
113             $E.notifyObservers(this, "hide");
114         },
115         /**
116          * 显示按钮
117          */
118         show: function(){
119             // shown in logic?
120             this._shownInLogic = true;
121 
122             // do nothing if it's unavailable
123             if (!this._available) {
124                 return;
125             }
126 
127             $D.show(this._node);
128             $E.notifyObservers(this, "show");
129         },
130         /**
131          * 设置按钮文本
132          * @param {String} text
133          */
134         setText: function(text){
135             this._node.innerHTML = $S.encodeHtml(text);
136         },
137         /**
138          * 设置按钮的title(鼠标移上去的tips)
139          * @param {String} title
140          */
141         setTitle : function(title){
142             this._node.title = $S.encodeScript(title);
143         },
144         /**
145          * 获取button dom对象
146          * @return {HTMLElement}
147          */
148         getNode: function(){
149             return this._node;
150         },
151         /**
152          * 设置按钮是否禁用
153          * @param {Boolean} isDisable
154          */
155         disable: function(isDisable){
156             isDisable = isDisable || false;
157             if(isDisable){
158                 $D.addClass(this._node, 'window_button_disabled');
159             }else{
160                 $D.removeClass(this._node, 'window_button_disabled');
161             }
162         }
163     });
164 });
