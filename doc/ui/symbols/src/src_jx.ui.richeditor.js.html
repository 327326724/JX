402     /**
403      * 在光标处插入一段html
404      * NOTE:调用时需确保光标在输入框中
405      * @param {String} html
406      */
407     insertHtml: function(html){
408         if(html === ''){
409             return;
410         }
411         var range = this.getRange();
412         if(!range){//如果拿不到输入框中的range, 就直接添加到最后
413             if(this.isEmpty()){
414                 this._divArea.innerHTML = html;
415             }else{
416                 this._divArea.innerHTML += html;
417             }
418             range = BaseEditor.getRange();
419             var divLastNode = this._divArea.lastChild;
420             if(range.selectNode){
421                 range.setEndAfter(divLastNode);
422                 range.setStartAfter(divLastNode);
423                 var selection = BaseEditor.getSelection();
424                 selection.removeAllRanges();
425                 selection.addRange(range);
426             }else if(range.moveToElementText){
427                 range.moveToElementText(divLastNode);
428                 range.collapse(false);
429                 range.select();
430             }
431         }else if(range.pasteHTML){//ie, ie9 也在这里
432 //            html += '<img style="display:inline;width:1px;height:1px;">';
433             range.pasteHTML(html);
434             range.collapse(false);
435             range.select();
436         }else if(range.createContextualFragment){//ie9竟然不支持这个方法
437             // 使用img标签是因为img是行内元素的同时, 又能设置宽高占位
438             html += '<img style="display:inline;width:1px;height:1px;">';
439             var fragment = range.createContextualFragment(html);
440             var lastNode = fragment.lastChild;
441             //如果已经选中了内容, 先把选中的删除
442             range.deleteContents();
443             range.insertNode(fragment);
444             //插入后把开始和结束位置都放到lastNode后面, 然后添加到selection
445             range.setEndAfter(lastNode);
446             range.setStartAfter(lastNode);
447             var selection = BaseEditor.getSelection();
448             selection.removeAllRanges();
449             selection.addRange(range);
450             //把光标滚动到可视区
451 //            if(lastNode.nodeType === 1){
452 //                ff开了firbug的时候, 会导致样式错乱, 换用scrollTop的方式
453 //                lastNode.scrollIntoView();
454 //            }
455             var divArea = this._divArea;
456             var pos = $D.getRelativeXY(lastNode, divArea);
457             divArea.scrollTop = pos[1] < divArea.scrollHeight ? divArea.scrollHeight : pos[1];
458             // 删除附加的节点, 这里只能这样删, chrome直接remove节点会导致光标消失掉
459             if(!$B.opera){//TODO opera的光标还原有问题, 要想办法解决
460                 document.execCommand('Delete', false, null);
461             }
462             if(BaseEditor.contains(divArea, lastNode)){//for opera
463                 divArea.removeChild(lastNode);
464             }
465         }
466         if(this.option.keepCursor){
467             //插入后把最后的range设置为刚刚的插入点
468             this.saveRange();
469         }
470     },
471     /**
472      * 往纯文本框插入一段文本
473      * @param {String} text
474      */
475     insertText: function(text){
476         if(text === ''){
477             return;
478         }
479         var textArea = this._textArea;
480         if($B.ie){
481             var range = BaseEditor.getRange();
482             if(range){
483                 range.text = text;
484             }else{
485                 textArea.value += text;
486             }
487         }else{
488             if(J.isUndefined(textArea.selectionStart)){
489                 textArea.value += text;
490             }else{
491                 var value = textArea.value,
492                     start = textArea.selectionStart, 
493                     end = textArea.selectionEnd,
494                     cursorPoint = start + text.length;
495                 textArea.value = value.substring(0, start) + text + value.substring(end);
496                 textArea.setSelectionRange(cursorPoint, cursorPoint);
497             };
498             
499         }
500     },

512     /**
513      * 清理节点, 把除了br,img之外的节点都清理掉
514      */
515     clearNodes: function(){
516         /* 
517          * 这里的原理是:
518          * 倒序遍历输入框的直接子节点
519          * 1. 如果是文本节点则跳过
520          * 2. 如果是element,且不是br,则用其包含的文本保存替换该节点
521          * 3. 如果是其他, 如comment,则移除
522          * 最后把光标位置还原
523          */
524         var divArea = this._divArea;
525         var text, textNode, cursorNode;
526         var childNodes = divArea.childNodes;
527 
528         for(var c = childNodes.length - 1, node; c >= 0; c--){
529             node = childNodes[c];
530             if(node.nodeType === 3){//text
531                 
532             }else if(node.nodeType === 1){//element
533                 if(node.nodeName !== 'BR'){
534                     if(this.option.nodeFilter && this.option.nodeFilter(node)){
535                         //nodeFilter返回true则不过滤
536                     }else{
537                         text = node.textContent || node.innerText || '';//innerText for ie
538                         if(text !== ''){
539                             textNode = document.createTextNode(text);
540                             if(!cursorNode){
541                                 cursorNode = textNode;
542                             }
543                             divArea.replaceChild(textNode, node);
544                         }else{
545                             divArea.removeChild(node);
546                         }
547                     }
548                     
549                 }
550             }else{//comment etc.
551                 divArea.removeChild(node);
552             }
553         }
554         if(cursorNode){//清除多余标签后还原光标位置
555             var selection = BaseEditor.getSelection();
556             if(selection.extend){//ff, chrome 要先扩展选区, 然后把选区开头合并到结尾
557                 //NOTE: chrome 拷贝某些html会有问题
558                 selection.extend(cursorNode, cursorNode.length);
559                 selection.collapseToEnd();
560             }
561         }
562     }

620         var keyCode = Number(e.keyCode);
621         if(keyCode === 16 || keyCode === 17 || keyCode === 18){
622             //排除掉单纯的shift,ctrl,alt键
623         }else if(this.option.keepCursor){
624             //延时进行保存, 避免连续输入文字的时候做了太多次操作
625             this._private.startTimeoutSaveRange(100);
626         }
627         this.onKeyUp(e);
628     },
629     onKeypress: function(e) {
630         this.onKeyUp(e);
631     },
632     onPaste: function(e){
633         if(this.option.clearNode){
634             //这里延时200毫秒, 如果onEditorPaste执行的时候也过滤的话,就可以取消这次了
635             this._private.startTimeoutClearNodes(200);
636         }
637     },
638     onDrop: function(e){
639         if(this.option.clearNode){
640             //因为发出这个事件的时候, 内容还没有粘贴到输入框, 所以要延时
641             this._private.startTimeoutClearNodes();
642         }
643     },
644     onEditorPaste: function(e){
645         if($E.notifyObservers(this, 'EditorPaste', e)){//如果EditorPaste的监听者返回了false, 则不进行过滤处理
646             if(this.option.clearNode){
647                 //因为发出这个事件的时候, 内容还没有粘贴到输入框, 所以要延时

649             }

667 /**
668  * 获取选中区, 如果传入了container, 则返回container的range
669  * @memberOf ui.BaseEditor
670  * @function
671  * @name getRange
672  * @param {HTMLElement} container  目标range的容器, 可选
673  * @return {Range}, null
674  */
675 BaseEditor.getRange = function(container) {
676     var selection = BaseEditor.getSelection();
677     if (!selection) {
678         return null;
679     }
680     var range = selection.getRangeAt ? (selection.rangeCount ? selection
681                 .getRangeAt(0) : null) : selection.createRange();
682     if(!range){
683         return null;
684     }
685     if(container){
686         if(BaseEditor.containsRange(container, range)){
687             return range;
688         }else{
689             return null;
690         }
691     }else{
692         return range;
693     }

697 /**
698  * 判断一个节点是否是某个父节点的子节点, 
699  * 默认不包含parent === child的情况
700  * @memberOf ui.BaseEditor
701  * @function
702  * @name contains
703  * @param {HTMLElement} parent
704  * @param {HTMLElement} child
705  * @param {Boolean} containSelf 指示是否可包含parent等于child的情况
706  * @return {Boolean} 包含则返回true
707  */
708 BaseEditor.contains = function(parent, child, containSelf){
709     if(!containSelf && parent === child){
710         return false;
711     }
712     if(parent.compareDocumentPosition){//w3c
713         var res = parent.compareDocumentPosition(child);
714         if(res == 20 || res == 0){
715             return true;
716         }
717     }else{
718         if(parent.contains(child)){//ie
719             return true;
720         }
721     }
722     return false;
723 };
724 /**
725  * 判断一个range是否被包含在container中
726  * @memberOf ui.BaseEditor
727  * @function
728  * @name containsRange
729  * @param {HTMLElement} container
730  * @param {Range} range
731  * @return {Boolean}
732  */
733 BaseEditor.containsRange = function(container, range){
734     var rangeParent = range.commonAncestorContainer || (range.parentElement && range.parentElement()) || null;
735     if(rangeParent){
736         return BaseEditor.contains(container, rangeParent, true);
737     }
738     return false;
739 };
