103      * 多行或单行字符串模板处理
104      * 
105      * @method template
106      * @memberOf string
107      * 
108      * @param {String} str 模板字符串
109      * @param {Object} obj 要套入的数据对象
110      * @return {String} 返回与数据对象合成后的字符串
111      * 
112      * @example
113      * <script type="text/html" id="user_tmpl">
114      *   <% for ( var i = 0; i < users.length; i++ ) { %>
115      *     <li><a href="<%=users[i].url%>"><%=users[i].name%></a></li>
116      *   <% } %>
117      * </script>
118      * 
119      * Jx().$package(function(J){
120      *  // 用 obj 对象的数据合并到字符串模板中
121      *  J.template("Hello, {name}!", {
122      *      name:"Kinvix"
123      *  });
124      * };

126     template = function(str, data){
127         // Figure out if we're getting a template, or if we need to
128         // load the template - and be sure to cache the result.
129         var fn = !/\W/.test(str) ?
130           cache[str] = cache[str] ||
131             template(document.getElementById(str).innerHTML) :
132           
133           // Generate a reusable function that will serve as a template
134           // generator (and which will be cached).
135           new Function("obj",
136             "var p=[],print=function(){p.push.apply(p,arguments);};" +
137             
138             // Introduce the data as local variables using with(){}
139             "with(obj){p.push('" +
140             
141             // Convert the template into pure JavaScript
142             str
143               .replace(/[\r\t\n]/g, " ")
144               .split("<%").join("\t")
145               .replace(/((^|%>)[^\t]*)'/g, "$1\r")
146               .replace(/\t=(.*?)%>/g, "',$1,'")
147               .split("\t").join("');")
148               .split("%>").join("p.push('")
149               .split("\r").join("\\'")
150           + "');}return p.join('');");
151         
152         // Provide some basic currying to the user
153         return data ? fn( data ) : fn;
154     };

180     /**
181      * 分解 URL 为一个对象，成员为：scheme, user, pass, host, port, path, query, fragment
182      * 
183      * @method parseURL
184      * @memberOf string
185      * 
186      * @param {String} str URL 地址
187      * @return {Object} 如果解析失败则返回 null
188      */
189     parseURL = function(str) {
190         var ret = null;
191 
192         if (null !== (ret = parseURL.RE.exec(str))) {
193             var specObj = {};
194             for (var i = 0, j = parseURL.SPEC.length; i < j ; i ++) {
195                 var curSpec = parseURL.SPEC[i];
196                 specObj[curSpec] = ret[i + 1];
197             }
198             ret = specObj;
199             specObj = null;
200         }
201 
202         return ret;
203     };

206     /**
207      * 将一个对象（成员为：scheme, user, pass, host, port, path, query, fragment）重新组成为一个字符串
208      * 
209      * @method buildURL
210      * @memberOf string
211      * 
212      * @param {Object} obj URl 对象
213      * @return {String} 如果是可接受的 url 则返回 true
214      */
215     buildURL = function(obj) {
216         var ret = '';
217 
218         // prefix & surffix
219         var prefix = {},
220             surffix = {};
221 
222         for (var i = 0, j = parseURL.SPEC.length; i < j ; i ++) {
223             var curSpec = parseURL.SPEC[i];
224             if (!obj[curSpec]) {
225                 continue;
226             }
227             switch (curSpec) {
228             case 'scheme':
229                 surffix[curSpec] = '://';
230                 break;
231             case 'pass':
232                 prefix[curSpec] = ':';
233             case 'user':
234                 prefix['host'] = '@';
235                 break;
236             //case 'host':
237             case 'port':
238                 prefix[curSpec] = ':';
239                 break;
240             //case 'path':
241             case 'query':
242                 prefix[curSpec] = '?';
243                 break;
244             case 'fragment':
245                 prefix[curSpec] = '#';
246                 break;
247             }
248 
249             // rebuild
250             if (curSpec in prefix) {
251                 ret += prefix[curSpec];
252             }
253             if (curSpec in obj) {
254                 ret += obj[curSpec];
255             }
256             if (curSpec in surffix) {
257                 ret += surffix[curSpec];
258             }
259         }
260 
261         prefix = null;
262         surffix = null;
263         obj = null;
264 
265         return ret;
266     };

274     /**
275      * 将 uri 的查询字符串参数映射成对象
276      * 
277      * @method mapQuery
278      * @memberOf string
279      * 
280      * @param {String} uri 要映射的 uri
281      * @return {Object} 按照 uri 映射成的对象
282      * 
283      * @example
284      * Jx().$package(function(J){
285      *  var url = "http://web.qq.com/?qq=4765078&style=blue";
286      *  // queryObj 则得到一个{qq:"4765078", style:"blue"}的对象。
287      *  var queryObj = J.mapQuery(url);
288      * };
289      */
290     mapQuery = function(uri){
291         //window.location.search
292         var i,
293             key,
294             value,
295             uri = uri || window.location.href,
296             index = uri.indexOf("?"),
297             pieces = uri.substring(index + 1).split("&"),
298             params = {};
299         if(index === -1){//如果连?号都没有,直接返回,不再进行处理. az 2011/5/11
300             return params;
301         }
302         for(i=0; i<pieces.length; i++){
303             try{
304                 index = pieces[i].indexOf("=");
305                 key = pieces[i].substring(0,index);
306                 value = pieces[i].substring(index+1);
307                 if(!(params[key] = unescape(value))){
308                     throw new Error("uri has wrong query string when run mapQuery.");
309                 }
310             }
311             catch(e){
312                 //J.out("错误：[" + e.name + "] "+e.message+", " + e.fileName+", 行号:"+e.lineNumber+"; stack:"+typeof e.stack, 2);
313             }
314         }
315         return params;
316     };
