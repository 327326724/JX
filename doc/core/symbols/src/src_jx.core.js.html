 74             /**
 75              * 【Jx 对象原型】
 76              * Jx
 77              * @class 
 78              * @constructor Jx
 79              * @global
 80              * 
 81              * @since version 1.0
 82              * @description Jx 对象原型的描述
 83              * 
 84              * @param {Number} ver 要使用的 Jx 的版本号，当前是1.0
 85              * @param {Boolean} isCreateNew 是否创建一个新的 Jx 实例，默认为 false 不创建新的 Jx 实例，只返回同一版本在全局中的唯一一个实例，注意：除非特殊需要，否则一般不要创建新的 Jx 实例
 86              * @return {Object} 返回对应版本的 Jx 对象
 87              * 
 88              * @example
 89              * //代码组织方式一(传统)：
 90              * var J = new Jx();
 91              * J.out(J.version);    //输出当前Jx的版本
 92              * 
 93              * @example
 94              * //代码组织方式二(推荐)：
 95              * Jx().$package(function(J){
 96              *     J.out(J.version);    //输出当前Jx的版本
 97              * };
 98              * //注：此种方式可以利用匿名函数来防止变量污染全局命名空间，尤其适合大型WebApp的构建！
 99              * 
100              * @example
101              * //范例：
102              * Jx().$package("tencent.alloy", function(J){
103              *     var $ = J.dom.id,
104              *     $D = J.dom,
105              *     $E = J.event,
106              *     $H = J.http;
107              *     this.name = "腾讯Q+ Web";
108              *     J.out(this.name);
109              * };
110              * 
111              */
112             Jx = function(ver, isCreateNew){
113                 var J = this;
114 
115                 if(isCreateNew){
116                     // 如果是第一次执行则初始化对象
117                     this._init();
118                 }else{
119                     if(ver){
120                         ver = String(ver);
121                         try{
122                             if(Jx.VERSIONS[ver]){
123                                 J = Jx.VERSIONS[ver];
124                             }else{
125                                 J = Jx.VERSIONS[Jx.DEFAULT_VERSION];
126                                 throw new Error("没有找到 JET version " + ver + ", 所以返回默认版本 JET version " + Jx.DEFAULT_VERSION + "!");
127                             }
128                         }catch(e){
129                             //J.out(e.fileName+";"+e.lineNumber+","+typeof e.stack+";"+e.name+","+e.message, 2);
130                             J.out("A.错误：[" + e.name + "] "+e.message+", " + e.fileName+", 行号:"+e.lineNumber+"; stack:"+typeof e.stack, 2);
131                         }
132                     }else{
133                         J = Jx.VERSIONS[Jx.DEFAULT_VERSION];
134                     }
135                 }
136                 return J;

170                 /**
171                  * 创建一个命名空间，创建的命名空间将会在 window 根命名空间下。
172                  * Create a new namespace, the top namespace is window.
173                  * 
174                  * @since version 1.0
175                  * @description 可以一次性连续创建命名空间
176                  * 
177                  * @param {String} name 命名空间名称
178                  * @returns {Object} 返回对最末命名空间的引用
179                  * 
180                  * @example
181                  * //在全局环境中创建tencent.alloy名字空间, $namespace完成的操作相当于在全局环境中执行如下语句：
182                  * //var tencent = {};
183                  * //tencent.alloy = {};
184                  * 
185                  * J.$namespace("tencent.alloy");
186                  * 
187                  * //注：Jx的$namespace方法与其他JS框架的namespace的方法不同，其他框架如YUI是在其YAHOO对像下创
188                  * //建命名空间，而Jx的$namespace测试直接在顶级命名空间window的下边直接创建命名空间。
189                  * 
190                  */
191                 $namespace: function(name) {
192                     var i,
193                         ni,
194                         nis = name.split("."),
195                         ns = window;
196 
197                     for(i = 0; i < nis.length; i=i+1){
198                         ni = nis[i];
199                         ns[ni] = ns[ni] || {};
200                         ns = ns[nis[i]];
201                     }
202 
203                     return ns;
204                 },

206                 /**
207                  * 创建一个 Javascript 代码包
208                  * 
209                  * @param {String} name 要创建的包的名字空间
210                  * @param {Function} func 要创建的包的包体
211                  * @returns {Mixed} 返回任何自定义的变量
212                  * 
213                  * @example
214                  * //创建一个匿名package包：
215                  * Jx().$package(function(J){
216                  *     //这时上下文对象this指向全局window对象
217                  *     alert("Hello world! This is " + this);
218                  * };
219                  * 
220                  * @example
221                  * //创建一个名字为tencent.kinvix的package包：
222                  * Jx().$package("tencent.kinvix", function(J){
223                  *     //这时上下文对象this指向window对象下的tencent.kinvix对象
224                  *     alert("Hello world! This is " + this);
225                  * };
226                  * 
227                  * 
228                  * 
229                  */
230                 $package: function(){
231                     var name = arguments[0],
232                         func = arguments[arguments.length-1],
233                         ns = topNamespace,
234                         returnValue;
235                         if(typeof func === "function"){
236                             if(typeof name === "string"){
237                                 ns = this.$namespace(name);
238                                 if(Jx.PACKAGES[name]){
239                                     //throw new Error("Package name [" + name + "] is exist!");
240                                 }else{
241                                        Jx.PACKAGES[name] = {
242                                         isLoaded: true,
243                                         returnValue: returnValue
244                                     };
245                                 }
246                                 ns.packageName = name;
247                             }else if(typeof name === "object"){
248                                 ns = name;
249                             }
250                             
251                             returnValue = func.call(ns, this);
252                         }else{
253                             throw new Error("Function required");
254                         }
255     
256                 },
