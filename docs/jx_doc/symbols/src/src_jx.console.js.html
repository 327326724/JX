615 		 * @return {String} 过滤后的结果

617 		filter : function(s){
618 			var reg = new RegExp(s,"i");
619 			var result = [];
620 			var logArr = this._log_record;
621 			J.array.forEach(logArr,function(log){
622 				var logStr = log.toString(true);
623 				if(reg.test(logStr)){
624 					result.push(log);
625 				}
626 			});
627 			return result;
628 		},
629 		/**
630 		 * 过滤type

634 		 */
635 		filterByType : function(logArr,type){
636 			var result = [];
637 			logArr = logArr||[];
638 			J.array.forEach(logArr,function(log){
639 				var isFound = false;
640 				J.array.forEach(type,function(i){
641 					if(log.type == i){
642 						isFound = true;
643 					}
644 				});
645 				if(isFound){
646 					result.push(log);
647 				}
648 			});
649 			return result;
650 		},
651 		/**

986 			function genTrace(ee, aa) {
987 				if (ee.stack) {
988 					return ee.stack;
989 				} else if (ee.message.indexOf("\nBacktrace:\n") >= 0) {
990 					var cnt = 0;

992 						cnt++;
993 						return (cnt % 2 == 0) ? "\n" : " @ ";
994 					});
995 				} else {
996 					var entry = (aa.callee == stack) ? aa.callee.caller : aa.callee;
997 					var eas = entry.arguments;
998 					var r = [];
999 					for (var i = 0, len = eas.length; i < len; i++) {
1000 						r.push((typeof eas[i] == 'undefined') ? ("<u>") : ((eas[i] === null) ? ("<n>") : (eas[i])));
1001 					}
1002 					var fnp = /function\s+([^\s\(]+)\(/;
1003 					var fname = fnp.test(entry.toString()) ? (fnp.exec(entry.toString())[1]) : ("<ANON>");
1004 					return (fname + "(" + r.join() + ");").replace(/\n/g, "");
1005 				}
1006 			}

1009 	
1010 			if ((e instanceof Error) && (typeof arguments == 'object') && (!!arguments.callee)) {
