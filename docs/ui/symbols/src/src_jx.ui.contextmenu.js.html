 65             var name = option.name || id;
 66             var parent = this._parent = option.container || (option.parentMenu ? option.parentMenu._parent : null) || document.body;
 67             var className = option.className || '';
 68             this.parentMenu = option.parentMenu;
 69             var context = this;
 70             var container = this._el = $D.id(id) || $D.node('div', {
 71                 id: id,
 72                 'class': 'context_menu',
 73                 style: 'display: none;'
 74             });
 75             var html = '<div class="context_menu_container "' + className + '"><ul id="'+ id +'_body" class="context_menu_item_list"></ul></div>';
 76 //                html += '<div class="context_menu_outer_r"></div><div class="context_menu_outer_b"></div>'

 79             if(J.browser.ie){//遮ie的flash
 80                 html += '<iframe class="context_menu_iframe" src="'+alloy.CONST.MAIN_URL+'domain.html"></iframe>';
 81             }
 82             container.innerHTML = html;
 83             parent.appendChild(container);
 84             
 85             if(option.width){
 86                 $D.setStyle(container, 'width', option.width + 'px');
 87             }
 88             this._body = $D.id(id + '_body');
 89             $E.off(container, 'contextmenu');//防止绑定多个事件
 90             $E.on(container, 'contextmenu', onSelfContextMenu);
 91             $E.addObserver(document, 'beforeStart', function(){
 92                 if(context.isShow()){
 93                     context.hide();
 94                 }
 95             });
 96             this._popupBox = new J.ui.PopupBox({
 97                 id: id,
 98                 name: name,
 99                 noCatchMouseUp: true,
100                 parentPopupBox: this.parentMenu ? this.parentMenu._popupBox : null,
101                 container: container
102             });
103             $E.addObserver(this._popupBox, 'hide', function(){
104                 context.hideSubmenu();
105                 $E.notifyObservers(context, 'onHide');
106             });
107             
108             this.setZIndex(topZIndex);
109             this._itemArr = [];
110             this._key2Item = {};
111             if(option.items){
112                 this._items_config= option.items;
113                 this.addItems(option.items);
114             }
115             if(option.triggers){
116                 var evnet = option.triggerEvent || 'contextmenu';

119                  */

288         /**
289          * 显示菜单
290          * @param {Number} x
291          * @param {Number} y
292          * @param {Number} offset 菜单位置相对于给定xy的偏移, 0为不偏移
293          * @param {HTMLElement} relativeEl 相对某一元素定位
294          */
295         show: function(x, y, offset, relativeEl){
296             $E.notifyObservers(this, 'BeforeShow', this);
297             var x = x || 0,
298                 y = y || 0,
299                 offset = typeof(offset) === 'undefined' ? 5 : offset;
300             var popup = this._popupBox;
301             var px = x + offset,
302                 py = y + offset;
303             var rw = 0, rh = 0, ie = J.browser.ie;
304             if(relativeEl){
305                 rw = $D.getOffsetWidth(relativeEl);
306                 rh = $D.getOffsetHeight(relativeEl);
307                 px += rw + 5;
308                 py -= 1;
309                 if(ie == 9 || ie == 8){
310                     px += 2;
311                 }
312             }
313             popup.setX('-10000');
314             popup.show();
315             var w = $D.getClientWidth(this._el),
316                 h = $D.getClientHeight(this._el),
317                 bodyWidth = $D.getClientWidth(),
318                 bodyHeight = $D.getClientHeight();
319             if(px + w > bodyWidth && x - w - offset > 0){
320                 if(rw){
321                     px = x - w - offset - 5;
322                     if(ie == 9 || ie == 8){
323                         px += 2;
324                     }
325                 }else{
326                     px = x - w - offset;
327                 }
328             }
329             if(py + h > bodyHeight && y - h - offset > 0){
330                 if(rh){
331                     py = y - h - offset + rh + 1;
332                 }else{
333                     py = y - h - offset;
334                 }
335             }
336             popup.setXY(px, py);
337             $E.notifyObservers(this, 'onShow', this);
338         },
339         /**
340          * 隐藏菜单
341          */
342         hide: function(){
343             this._popupBox.hide();
344             $E.notifyObservers(this, 'onHide', this);
345         },
346         /**
347          * 隐藏所有子菜单
348          */
349         hideSubmenu: function(){
350             for(var i in this._itemArr){
351                 if(this._itemArr[i].getSubmenu){
352                     this._itemArr[i].getSubmenu().hide();
353                 }
354             }
355         },
356         /**
357          * 指示菜单是否显示
358          * @return {Boolean}
359          */
360         isShow: function(){
361             return this._popupBox.isShow();
362         },
363         /**
364          * 销毁菜单
365          */
366         destory: function(){
367             this._el.innerHTML = '';
368             $E.off(this._el, 'contextmenu');
369             this.clearItems();
370             for (var p in this) {
371                 if (this.hasOwnProperty(p)) {
372                     delete this[p];
373                 }
374             }
375         }

460                 /**

462                  */
463                 onClickFunc = function(e){
464                     e.preventDefault();

501

830     /**
831      * 【ContextSubmenuItem】子菜单项, 关联着一个子菜单
832      * 
833      * @class 
834      * @name ContextSubmenuItem
835      * @augments  ui.ContextMenuItem
836      * @constructor
837      * @memberOf ui
838      * @param {Object} option 参数对象
839      * {
840      * title: text,
841      * text: '',
842      * link: 'javascript:void(0);',
843      * icon: null,
844      * items: null,//子菜单的菜单项, 必选项
845      * subWidth: null,//子菜单的宽
846      * subContainer: null //子菜单的容器
847      * }
848      * icon 为object, 包括{url,className,style}其中一个
849      * 
850      * @since version 1.0
851      * @description
852      */
853     var ContextSubmenuItem = J.ui.ContextSubmenuItem = new J.Class({extend: ContextMenuItem}, 
854     /**
855      * @lends ui.ContextSubmenuItem.prototype
856      */
857     {
858         /**
859          * @ignore
860          */
861         init: function (option){
862             if(!option.items){
863                 throw new Error('J.ui.ContextSubmenuItem: option.items is null!');
864             }
865             option.title = option.title || option.text || '';
866             var defaultOption = {
867                 title: null,
868                 text: '',
869                 autoHide: true,
870                 link: 'javascript:void(0);',
871                 icon: null,
872                 enable: true,
873                 subWidth: null,
874                 parentMenu: option.parentMenu
875             };
876             delete option.parentMenu;//不删除的话, extend会导致parentMenu指针不对
877             option = this.option = J.extend(defaultOption, option);
878             this.parentMenu = option.parentMenu;
879             var liNode = this._el = $D.node('li', {
880                 'class': 'context_menu_item_container'
881             });
882             this.render();
883             if(option.enable){
884                 this.enable();
885             }else{
886                 this.disable();
887             }
888             var subOption = {
889                 parentMenu: option.parentMenu,
890                 width: option.subWidth,
891                 beforeShow: option.beforeShow,
892                 items: option.items
893             };
894             this._submenu = new ContextMenu(subOption);
895                 
896             var onClickFunc, context = this;
897             
898             var submenuTimer = context.sunmenuTimer = 0, submenuHideTimeout = 200;
899             /**
900              * @ignore
901              */
902             var hideSubmenu = function(){
903                 if(context._submenu.isShow()){
904                     context._submenu.hide();
905                 }
906             };
907             var observer = {
908                 /** @ignore */
909                 onItemMouseenter: function(e){
910                     e.stopPropagation();
911                     if(context._isEnable){
912                         var xy = $D.getClientXY(this);
913                         context._submenu.setZIndex(context.parentMenu.getZIndex());
914                         context._submenu.show(xy[0], xy[1], 0, this);
915                     }
916                 },
917                 /** @ignore */
918                 onItemMouseleave: function(e){
919                     if(submenuTimer){
920                         window.clearTimeout(submenuTimer);
921                         submenuTimer = 0;
922                     }
923                     submenuTimer = window.setTimeout(hideSubmenu, submenuHideTimeout);
924                 },
925                 /** @ignore */
926                 onSubmenuMouseenter: function(e){
927                     if(submenuTimer){
928                         window.clearTimeout(submenuTimer);
929                         submenuTimer = 0;
930                     }
931                     $D.addClass(liNode, 'context_menu_item_hover');
932                 },
933                 /** @ignore */
934                 onSubmenuMouseleave: function(e){
935                     observer.onItemMouseleave(e);
936                     //$D.removeClass(liNode, 'context_menu_item_hover');
937                 },
938                 /** @ignore */
939                 onSubmenuShow: function(){
940                     $D.addClass(liNode, 'context_menu_item_hover'); 
941                 },
942                 /** @ignore */
943                 onSubmenuHide: function(){
944                     $D.removeClass(liNode, 'context_menu_item_hover');
945                 }
946             };
947             
948             var submenuEl = context._submenu.getElement();
949             $E.on(liNode, 'mouseenter', observer.onItemMouseenter);
950             if(option.autoHide){
951                 $E.on(liNode, 'mouseleave', observer.onItemMouseleave);
952                 $E.on(submenuEl, 'mouseenter', observer.onSubmenuMouseenter);
953                 $E.on(submenuEl, 'mouseleave', observer.onSubmenuMouseleave);
954             }
955             //$E.addObserver(context._submenu, 'onShow', observer.onSubmenuShow);
956             $E.addObserver(context._submenu, 'onHide', observer.onSubmenuHide);
957             if(option.onClick){
958                 /**
959                  * @ignore
960                  */
961                 onClickFunc = function(e){
962                     e.preventDefault();
963                     if(context._isEnable){
964                         option.onClick.call(this, e, context);
965                         observer.onItemMouseenter.call(this, e);
966                     }
967                 }
968             }else{
969                 /**
970                  * @ignore
971                  */
972                 onClickFunc = function(e){
973                     e.preventDefault();
974                     observer.onItemMouseenter.call(this, e);
975                 }
976             }
977             $E.on(liNode, 'click', onClickFunc);
978         },
979         /**
980          * 获取该菜单项关联的子菜单
981          */
982         getSubmenu: function(){
983             return this._submenu;
984         },
985         /**
986          * 绘制菜单项的内容
987          */
988         render: function(){
989             var option = this.option;
990             var html = '<a class="context_menu_item" href="' + option.link + '" title="' + option.title + '">';
991             if(option.icon){
992                 var icon = option.icon,
993                     image = icon.url ? 'background-image: url(' + icon.url + ');' : '',
994                     style = (icon.style || '') + image,
995                     cls = icon.className || '';
996                 html += '<span class="context_menu_item_icon '+ cls +'" style="' + style + '"></span>';
997             }
998             html += '<span class="context_menu_item_text">' + option.text + '</span><span class="context_menu_item_subicon"></span></a>';
999             this._el.innerHTML = html;
1000         }
1001     });
