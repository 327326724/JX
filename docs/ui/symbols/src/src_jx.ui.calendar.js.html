 65         function HS_DateAdd(interval,number,date){
 66             number = parseInt(number);
 67             if (typeof(date)=="string"){
 68                 var date = new Date(date.split("-")[0],date.split("-")[1],date.split("-")[2]);
 69             }
 70             if (typeof(date)=="object"){
 71                 var date = date;
 72             }
 73             switch(interval){
 74                 case "y":return new Date(date.getFullYear()+number,date.getMonth(),date.getDate()); break;
 75                 case "m":return new Date(date.getFullYear(),date.getMonth()+number,HS_checkDate(date.getFullYear(),date.getMonth()+number,date.getDate())); break;
 76                 case "d":return new Date(date.getFullYear(),date.getMonth(),date.getDate()+number); break;
 77                 case "w":return new Date(date.getFullYear(),date.getMonth(),7*number+date.getDate()); break;
 78             }
 79         }

 81         function HS_checkDate(year,month,date){
 82             var enddate = ["31","28","31","30","31","30","31","31","30","31","30","31"];
 83             var returnDate = "";
 84             if (year%4==0){
 85                 enddate[1]="29";
 86             }
 87             if (date>enddate[month]){
 88                 returnDate = enddate[month];
 89             }else{
 90                 returnDate = date;
 91             }
 92             return returnDate;
 93         }

 95         function HS_WeekDay(date){
 96             var theDate;
 97             if (typeof(date)=="string"){
 98                 theDate = new Date(date.split("-")[0],date.split("-")[1],date.split("-")[2]);
 99             }
100             if (typeof(date)=="object"){
101                 theDate = date;
102             }
103             return theDate.getDay();
104         }
105         //设置当前日期对象(根据选择动态变化)
106         function setNow( date ){
107             var now;
108             if (typeof(arguments[0])=="string"){
109                 var selectDate = arguments[0].split("-");
110                 var year = selectDate[0];
111                 var month = parseInt(selectDate[1])-1+"";
112                 var date = selectDate[2];
113                 now = new Date(year,month,date);
114             }else if (typeof(arguments[0])=="object"){
115                 now = arguments[0];
116             }
117             _now.year = now.getFullYear();
118             _now.month = now.getMonth();
119             _now.date = now.getDate();
120         }
121         function HS_calender(){
122             var lis = "";            
123             setNow(arguments[0]); 
124             var yearMonthStr = _now.year+"-"+_now.month;    
125             var lastMonthEndDate = HS_DateAdd("d","-1",yearMonthStr+"-01").getDate();
126             var lastMonthDate = HS_WeekDay(yearMonthStr+"-01");
127             var thisMonthLastDate = HS_DateAdd("d","-1",_now.year+"-"+(parseInt(_now.month)+1).toString()+"-01");
128             var thisMonthEndDate = thisMonthLastDate.getDate();
129             var thisMonthEndDay = thisMonthLastDate.getDay();
130             
131             var today = _today.getFullYear()+"-";
132             var month = _today.getMonth() + 1;
133             today += month < 10 ? '0' : '';
134             today += month.toString() + '-';
135             today += _today.getDate() < 10 ? '0'  : '';
136             today += _today.getDate().toString();
137              //生成日期内容：可能存在三部分：上个月，当前月，下个月的       
138              //上个月
139             var i;
140             for (i=0; i<lastMonthDate; i++){ // Last Month's Date
141                 lis = "<li class='lastMonthDate'>"+lastMonthEndDate+"</li>" + lis;
142                 lastMonthEndDate--;
143             }
144             //当前月   
145             var month = parseInt(_now.month)+1;
146             month = month < 10 ? '0' + month.toString() : month;
147             
148             for (i=1; i<=thisMonthEndDate; i++){ // Current Month's Date
149                 var date = i < 10 ? '0' + i.toString() : i;
150                 var dateStr = _now.year + "-" + month + "-" + date;
151                 var checkHight = checkHightLight(dateStr); 
152                 var hightClass = checkHight ? ' class="have"' : '';             
153                 var todayClass = today == dateStr ? ' class="today"' : '';  
154                 if( _isOnlyHight ){
155                     if( checkHight ){
156                         lis += "<li "+hightClass+"><a href='###' "+todayClass+" title='"+dateStr+"'>"+i+"</a></li>";    
157                     }else{
158                         lis += "<li "+hightClass+">"+i+"</li>";     
159                     }
160                 }else{
161                     lis += "<li "+hightClass+"><a href='###' "+todayClass+" title='"+dateStr+"'>"+i+"</a></li>";        
162                 }
163             }
164             //下个月
165             var j=1;
166             for (i=thisMonthEndDay; i<6; i++){ // Next Month's Date
167                 lis += "<li class='nextMonthDate'>"+j+"</li>";
168                 j++;
169             }
170              
171             //生成标题
172             var CalenderTitle = "<a href='###' class='nextMonth' title='下一个月'>»</a>";
173             CalenderTitle += "<a href='###' class='lastMonth' title='上一个月'>«</a>";
174             CalenderTitle += "<span class='selectThisYear'><a href='###' class='toSelectYear' title='点击这里选择年份' >"+
175                               _now.year+"</a></span>年<span class='selectThisMonth'>"+
176                               "<a href='###' class='toSelectMonth' title='点击这里选择月份'>"+
177                               (parseInt(_now.month)+1).toString()+"</a></span>月";
178             if (arguments.length>1){                
179                 _dom.date.innerHTML = lis;
180                 _dom.title.innerHTML =  CalenderTitle;
181                 bindTitleClick();
182                 bindDateClick();
183                 bindSelectClick();
184             }else{
185                 //日历内容框架
186                 var CalenderBox ="<div id='calender' class='calender'>\
187                                     <div class='calenderTitle'>"+CalenderTitle+"</div>\
188                                     <div class='calenderBody'>\
189                                         <ul class='day'>\
190                                             <li>日</li><li>一</li><li>二</li><li>三</li><li>四</li><li>五</li><li>六</li>\
191                                         </ul>\
192                                         <ul class='date' id='thisMonthDate'>"+lis+"</ul>\
193                                     </div>\
194                                     <div class='calenderBottom'>\
195                                         <span><a href='###' class='closeCalender' >关闭</a><span></span>\
196                                         <span><a href='###' class='calToToday' title='"+dateStr+"'>今天</a></span>\
197                                         <span><a href='###' class='calEmptyInput' title=''>置空</a></span>\
198                                     </div>\
199                                   </div>";
200                                   
201                 return CalenderBox;
202             }
