  2  Jx().$package(function (J) {
  3      var $D = J.dom,
  4          $E = J.event;
  5         
  6     J.ui = J.ui || {};
  7     /**
  8      * 自定义滚动条
  9      * @class
 10      * @constructor
 11      * @memberOf ui
 12      * @name ScrollBar
 13      * @param {HTMLElement} el 需要生成滚动条的div(至少保证有一个子div，存放需要滚动的内容)
 14      * @param {Object} option
 15      * option.onscroll||function(){};<br/>
 16        option.scrollToBottom||function(){};<br/>
 17        option.ipadTouchArea||false;<br/>
 18      * @example 
 19      * <div id ="scroll"><div id="scrollcontent"></div></div>
 20      * scroll与scrollcontent的position必须为relative|absolute
 21      * (可选: 会用js强制设置)scrollcontent的高度为100%, overflow:hidden
 22      */
 23     J.ui.ScrollBar = new J.Class(
 24     /**
 25      * @lends ui.ScrollBar.prototype
 26      */
 27     {
 28         option: {
 29             'barClass': 'scrollBar',
 30             'barHoverClass': null,
 31             'barActiveClass': null,
 32             'showBarContainer': false
 33         },
 34         /**
 35          * @ignore
 36          */
 37         init: function(el,option) {
 38             
 39             var self = this;
 40             J.extend(this.option, option);
 41             this.bar = $D.node('div',{
 42                 'class':option.scrollBarClassName||'scrollBar' 
 43             });
 44             this.obj = el;
 45             this.content = this.obj.getElementsByTagName('div')[0] || this.obj;
 46             $D.setStyle(this.content, 'height', '100%');
 47             $D.setStyle(this.content, 'overflow', 'hidden');
 48 
 49             //兼容ie圆角
 50             if(J.browser.ie){
 51                 this.bar.innerHTML = '<div class="scrollBar_bg scrollBar_bg_t"></div><div class="scrollBar_bg scrollBar_bg_b"></div>';
 52             }
 53             this.onscroll= option.onscroll||function(){};
 54             this.scrollToBottom= option.scrollToBottom||function(){};
 55             this.ipadTouchArea= option.ipadTouchArea||false;
 56             $D.setStyle(this.bar, 'marginTop', 0);
 57             this.obj.appendChild(this.bar);
 58             
 59             //滚动槽
 60             if(option.showBarContainer){
 61                 this.barBc = $D.node('div',{
 62                     'class':'scrollBar_bgc'
 63                 });
 64                 if(J.browser.ie){
 65                     this.barBc.innerHTML = '<div class="scrollBar_bgc_c scrollBar_bgc_t"></div><div class="scrollBar_bgc_c scrollBar_bgc_b"></div>';
 66                 }
 67                 this.obj.appendChild(this.barBc);
 68             }
 69             
 70             this.setBarHeight();
 71             this.bar.y;
 72             this.srcElement;
 73             this.marginTop;
 74             this.D; //鼠标滚动方向
 75             this.wheelThread = 20; //滚动20px
 76             this.isScrolling = false;
 77             
 78             var observer = {
 79                 /** @ignore */
 80                 onMouseDown:function(e){
 81                     //e.preventDefault();
 82                     var target = e.target;
 83                     if(e.changedTouches) {
 84                         e= e.changedTouches[0];
 85                     }
 86                     self.bar.y = e.clientY;
 87                     self.bar.t = parseInt( self.bar.style.marginTop );
 88                     if(!J.platform.iPad){
 89                         $E.on(document, 'mousemove', observer.onMouseMove);
 90                         $E.on(document, 'mouseup', observer.onMouseUp);
 91                         e.stopPropagation();
 92                     }else {
 93                         if(target == self.bar){
 94                             $E.on(document, 'touchmove', observer.onMouseMove);
 95                             $E.on(document, 'touchend', observer.onBarMouseUp);
 96                         }else{
 97                             $E.on(document, 'touchmove', observer.onTouchAreaMove);
 98                             $E.on(document, 'touchend', observer.onMouseUp);
 99                         }
100                     }
101                     self.isScrolling = true;
102                     if(self.option.barActiveClass)
103                         $D.addClass(self.bar, self.option.barActiveClass);
104                 },
105                 /** @ignore */
106                 onMouseMove:function(e){
107                     if(e.changedTouches) {
108                         e.preventDefault();
109                         e= e.changedTouches[0];
110                     }
111                     self.scroll( e.clientY - self.bar.y );
112                     if(!J.platform.iPad) {
113                         e.preventDefault();
114                         e.stopPropagation();
115                     }
116                 },
117                 /** @ignore */
118                 onTouchAreaMove:function(e){
119                     e.preventDefault();
120                     e.stopPropagation();
121                     if(e.changedTouches) {
122                         e= e.changedTouches[0];
123                     }
124                     self.scroll( - e.clientY + self.bar.y );
125                     //if(!J.platform.iPad) {
126                     //}
127                 },
128                 /** @ignore */
129                 onBarMouseUp:function(e){
130                     if(!J.platform.iPad){
131                         $E.off(document, 'mousemove', observer.onMouseMove);
132                         $E.off(document, 'mouseup', observer.onMouseUp);
133                     }else {
134                         $E.off(document, 'touchmove', observer.onMouseMove);
135                         $E.off(document, 'touchend', observer.onBarMouseUp);
136                     }
137                     self.isScrolling = false;
138                     if(self.option.barActiveClass)
139                         $D.removeClass(self.bar, self.option.barActiveClass);
140                 },
141                 /** @ignore */
142                 onMouseUp:function(e){
143                     if(!J.platform.iPad){
144                         $E.off(document, 'mousemove', observer.onMouseMove);
145                         $E.off(document, 'mouseup', observer.onMouseUp);
146                     }else {
147                         $E.off(document, 'touchmove', observer.onTouchAreaMove);
148                         $E.off(document, 'touchend', observer.onMouseUp);
149                     }
150                     self.isScrolling = false;
151                     if(self.option.barActiveClass)
152                         $D.removeClass(self.bar, self.option.barActiveClass);
153                 },
154                 /** @ignore */
155                 onMouseOver: function(e){
156                     $D.addClass(self.bar, self.option.barHoverClass);
157                 },
158                 /** @ignore */
159                 onMouseOut: function(e){
160                     $D.removeClass(self.bar, self.option.barHoverClass);
161                 },
162                 /** @ignore */
163                 onMouseWheel:function(event){
164                     if(!$D.isShow(self.bar)) {
165                         self.scrollToBottom(event);
166                         return;
167                     }
168                     self.D = event.wheelDelta;
169                     event.returnValue = false;
170                     var _y = (self.D < 0) ? self.wheelThread : (0-self.wheelThread);
171                     self.bar.y = event.clientY;
172                     self.bar.t = parseInt( self.bar.style.marginTop );
173 //                    var mul = self.content.scrollHeight/parseInt(self.bar.style.height)/5;
174 //                    if(mul<1){
175 //                        mul = 1;
176 //                    }
177 //                    _y *= mul;
178                     self.scroll(_y);
179                 },
180                 /** @ignore */
181                 onClick:function(e){
182                     e.stopPropagation();
183                 },
184                 /** @ignore */
185                 onDomMouseScroll:function(e){
186                     if(!$D.isShow(self.bar)) {
187                         self.scrollToBottom(e);
188                         return;
189                     }
190                     self.D = (e.detail > 0) ? -1 : 1;
191                     if(!J.platform.iPad){
192                         e.stopPropagation();
193                         e.preventDefault();
194                     }
195                     self.bar.y = e.clientY;
196                     self.bar.t = parseInt( self.bar.style.marginTop );
197                     var _y = (self.D < 0) ? self.wheelThread : (0-self.wheelThread);
198 //                    var mul = self.content.scrollHeight/parseInt(self.bar.style.height)/5;
199 //                    if(mul<1){
200 //                        mul = 1;
201 //                    }
202 //                    _y *= mul;
203                     self.scroll(_y);
204                 }
205             };
206             
207             if(this.option.stopClick){
208                 $E.on(this.bar, 'click', observer.onClick);
209             }
210             if(this.option.barHoverClass){
211                 $E.on(this.bar, 'mouseover', observer.onMouseOver);
212                 $E.on(this.bar, 'mouseout', observer.onMouseOut);
213             }
214             if(J.platform.iPad) {
215                 $E.on(this.bar, 'touchstart', observer.onMouseDown);
216                 if(this.ipadTouchArea) {
217                     $E.on(this.content, 'touchstart', observer.onMouseDown);
218                 }
219             }else {
220                 $E.on(this.bar, 'mousedown', observer.onMouseDown);
221                 //滚动事件兼容 参照jquery mousewheel
222                 if (J.browser.ie || J.browser.engine.webkit || J.browser.opera) {
223                     $E.on(this.content, 'mousewheel', observer.onMouseWheel);
224                     $E.on(this.bar, 'mousewheel', observer.onMouseWheel);
225                 } else {
226                     $E.on(this.content, 'DOMMouseScroll', observer.onDomMouseScroll);
227                     $E.on(this.bar, 'DOMMouseScroll', observer.onDomMouseScroll);
228                 }
229             }
230         },
231         /**
232          * 滚动条回到顶部
233          */
234         scrollBack: function() {
235             var self= this;
236             self.content.scrollTop = "0px";
237             self.bar.t= 0;
238             self.scroll(0);
239         },
240         /**
241          * 刷新滚动条, 同 update
242          */
243         refresh: function(){
244             //跟iscroll同名接口
245             this.update();  
246         },
247         /**
248          * 刷新滚动条
249          */
250         update: function(){
251             this.setBarHeight();
252         },
253         /**
254          * 计算滚动条位置
255          * @private
256          */
257         setBarHeight: function(){
258             
259             var self = this;
260             
261             self.onscroll(0,0); //防止影响scrollHeight计算
262             self.bar.style['height']='0'; //防止影响scrollHeight计算
263             
264             $D.hide(self.bar); //防止影响scrollHeight计算
265             
266             if((self.content.offsetHeight - self.content.scrollHeight) >= 0 ){
267                 if(self.barBc){
268                     $D.hide(self.barBc);
269                 }
270                 self.bar.t = 0; //滚动条复位
271             }else{
272                 self.bar.style.height = parseInt(self.content.offsetHeight / self.content.scrollHeight * self.content.offsetHeight) + 'px';
273                 $D.show(self.bar);
274                 if(self.barBc){
275                     $D.show(self.barBc);
276                 }
277                 self.bar.t = parseInt( self.bar.style.marginTop );
278             }
279             //触发scroll事件，重置滚动条位置，滚动距离为0
280             self.scroll(0);
281         },
282         /**
283          * 移动滚动条，dis为滚动条移动距离
284          * @param {Number} dis 
285          */
286         scroll: function(dis){
287             var self = this;
288             self.marginTop = (self.bar.t||0) + dis;
289             if( self.marginTop < 0 ) {
290                 self.marginTop = 0;
291             }
292             if( self.marginTop > self.content.clientHeight - self.bar.offsetHeight ){
293                 self.marginTop = self.content.clientHeight - self.bar.offsetHeight;
294                 //到底通知
295                 self.scrollToBottom();
296             }else {
297                 
298             }
299             self.bar.style.marginTop = self.marginTop + 'px';
300             if(dis==0) {
301                 self.onscroll(dis,dis);
302             }
303             var scrollTop= ( self.content.scrollHeight - self.content.offsetHeight ) * parseInt( self.marginTop ) / ( self.content.offsetHeight - self.bar.offsetHeight );
304             self.content.scrollTop = scrollTop;
305             self.onscroll(scrollTop,dis);
306             
307         },
308         /**
309          * 获取当前滚动条的位置
310          * @return {Number}
311          */
312         getScrollTop : function(){
313           return  parseInt(this.content.scrollTop);
314         },
315         /**
316          * 移动面板内容，dis为面板移动距离
317          * @param {Number} dis 
318          */
319         contentScroll: function(dis){
320             var self = this;
321             var dis = parseInt(self.obj.offsetHeight / self.content.scrollHeight * dis);
322             this.scroll(dis);
323         },
324         /**
325          * 获取当前内容区的滚动位置
326          * @return {Number}
327          */
328         contentPosition: function() {
329             return this.content.scrollTop;
330         }
331     });
332 });
