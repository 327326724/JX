 15 Jet().$package(function (J) {
 16     var $D = J.dom,
 17         $S = J.string,
 18         $E = J.event;
 19         
 20     J.ui = J.ui || {};
 21     /**
 22      * 带箭头的气泡类
 23      * @class
 24      * @memberOf ui
 25      * @name Bubble
 26      * @constructor
 27      * @param {Object} option 构造参数
 28      * {
 29      *   bubbleParent, {HTMLElement} 气泡的dom的父节点
 30      * }
 31      * @description 
 32      * 注意, bubble的自动关闭都是hide调用, 是不会把自己删除的, 要删除bubble, 调用close方法
 33      * 
 34      */
 35     var Bubble = J.ui.Bubble = J.Class(
 36     /**
 37      * @lends ui.Bubble.prototype
 38      */
 39     {
 40         /**
 41          * 初始化气泡
 42          * @private
 43          * @ignore
 44          */
 45         init: function(option){
 46             option = option || {};
 47             var defaultOption = {
 48                 bubbleParent: document.body,
 49                 className: '',
 50                 hasCloseButton: true,
 51                 closeOnHide: false,
 52                 zIndex: 1000000
 53             };
 54             option = this.option = J.extend(defaultOption, option);
 55             var id = this._getId();
 56             
 57             var html = '<div id="bubble_tip_pointer_' + id + '" class="bubble_tip_pointer bubble_tip_pointer_left"></div>\
 58                 <div class="bubble_tip_head"></div>\
 59                 <div class="bubble_tip_body">\
 60                     <div class="bubble_tip_title"><a id="bubble_tip_close_' + id + '" href="###" class="bubble_tip_close" title="关闭">x</a>\
 61                         <span id="bubble_tip_title_' + id + '"></span></div>\
 62                     <div id="bubble_tip_content_' + id + '" class="bubble_tip_content"></div>\
 63                 </div>\
 64                 <div id="bubble_tip_foot_' + id + '" class="bubble_tip_foot">\
 65                     <a id="bubble_tip_btn_next_' + id + '" href="###" class="bubble_tip_btn"></a>\
 66                     <a id="bubble_tip_btn_ok_' + id + '" href="###" class="bubble_tip_btn"></a>\
 67                 </div>\
 68                 <iframe width="100%" height="100%" class="bubble_tip_bg_iframe" src="about:blank"></iframe>';
 69             var divNode = $D.node("div",{
 70                 'class': 'bubble_tip_container ' + option.className
 71             });
 72             divNode.innerHTML = html;
 73             $D.setCssText(divNode, 'left: -10000px; top: 0px; z-index: ' + option.zIndex + ';');
 74             option.bubbleParent.appendChild(divNode);
 75             
 76             
 77             this._container = divNode;
 78             this._title = $D.id('bubble_tip_title_' + id);
 79             this._content = $D.id('bubble_tip_content_' + id);
 80             this._pointer = $D.id('bubble_tip_pointer_' + id);
 81             this._okBtn = $D.id('bubble_tip_btn_ok_' + id);
 82             this._nextBtn = $D.id('bubble_tip_btn_next_' + id);
 83             this._closeBtn = $D.id('bubble_tip_close_' + id);
 84             if(!option.hasCloseButton){
 85                 $D.hide(this._closeBtn);
 86             }
 87             var context = this;
 88             
 89             var observer = {
 90                 /**
 91                  * @ignore
 92                  */
 93                 onCloseBtnClick: function(e){
 94                     e.preventDefault();
 95                     e.stopPropagation();
 96                     $E.notifyObservers(context, 'onBubbleClose', context);
 97                     context.hide();
 98                 },
 99                 /**
100                  * @ignore
101                  */
102                 onOkButtonClick: function(e){
103                     e.preventDefault();
104                     e.stopPropagation();
105                     $E.notifyObservers(context, 'onBubbleOkBtnClick', context);
106                     context.hide();
107                 },
108                 /**
109                  * @ignore
110                  */
111                 onNextButtonClick: function(e){
112                     e.preventDefault();
113                     e.stopPropagation();
114                     $E.notifyObservers(context, 'onBubbleNextBtnClick', context);
115                 }
116             };
117             
118             $E.on(this._closeBtn, 'click', observer.onCloseBtnClick);
119             $E.on(this._okBtn, 'click', observer.onOkButtonClick);
120             $E.on(this._nextBtn, 'click', observer.onNextButtonClick);
121             
122         },
123         /**
124          * 获取容器的Dom元素
125          * @return {HTMLElement}
126          */
127         getElement: function(){
128             return this._container;
129         },
130         /**
131          * 显示气泡
132          * @param {Object} option
133          * @example
134          * option 的默认参数为:
135          * {
136          *       pointerPosition: 'top right',//箭头的位置
137          *       pointerOffset: 20,//箭头在气泡边上的偏移值
138          *       pointerSize: [18, 12],//箭头的高度/宽度
139          *       position: [0,0],//气泡的位置,直接设置position的时候,不要设置target,否则不起作用
140          *       target: null,//{HTMLElement} 气泡所指的dom节点, ps: 如果该属性存在, 则忽略position的设置
141          *       targetOffset: [x, y]//气泡位置的便宜, x/y可正负 
142          *  };
143          *  
144          * pointerPosition的值可为
145          *  "top left"
146          *  "top right"
147          *  "bottom left"
148          *  "bottom right"
149          *  "left top"
150          *  "left bottom"
151          *  "right top"
152          *  "right bottom"
153          *  
154          *  @description
155          *  TODO 后续可添加通过设置targetPosition来直接指定箭头以及气泡的位置
156          */
157         show: function(option){
158             option = option || {};
159             var defaultOption = {
160                 pointerPosition: 'top right',
161                 pointerOffset: 20,
162                 pointerSize: [18, 12],
163                 position: [0, 0],
164                 target: null,
165                 targetOffset: [0, 0]
166             };
167             option = J.extend(defaultOption, option);
168             //检查箭头的位置参数是否合法
169             if(!this._checkPointerPosition(option.pointerPosition)){
170                 throw new Error('Bubble >>>> the pointerPosition\'s value is not correct');
171             }
172             //设置箭头在bubble内的位置
173             this._setPointerPosition(option.pointerPosition, option.pointerOffset);
174             //设置气泡的位置
175             this._setBubblePosition(option);
176             
177             $D.show(this._container);
178         },
179         /**
180          * 设置气泡的z-index
181          * @param {Int} zIndex
182          */
183         setZIndex: function(zIndex){
184             $D.setStyle(this._container, 'zIndex', zIndex);  
185         },
186         /**
187          * 设置气泡外框的样式
188          * @param {String} property 属性
189          * @param {String} value 值
190          */
191         setContainerStyle: function(property, value){
192             $D.setStyle(this._container, property, value);
193         },
194         /**
195          * 设置标题文字
196          * @param {String} text 标题文字
197          */
198         setTitle: function(text){
199             this._title.innerHTML = text;
200         },
201         /**
202          * 设置主题内容
203          * @param {String} html 主题内容, 可包含html标签
204          */
205         setContent: function(html){
206             this._content.innerHTML = html;  
207         },
208         /**
209          * 显示一个的按钮
210          * @param {String} type 按钮的类型
211          * @param {String} text 按钮的文字
212          * @param {Boolean} highlight 是否显示高亮的按钮
213          */
214         showButton: function(type, text, highlight){
215             var btn = this['_' + type + 'Btn'];
216             highlight = J.isUndefined(highlight) ? false : true;
217             if(btn){
218                 btn.innerHTML = text;
219                 $D.show(btn);
220                 if(highlight){
221                     $D.addClass(btn, 'bubble_tip_btn_next');
222                 }else{
223                     $D.addClass(btn, 'bubble_tip_btn_ok');
224                 }
225             }
226             return btn;
227         },
228         /**
229          * 隐藏按钮
230          * @param {String} type 按钮类型
231          */
232         hideButton: function(type){
233             var btn = this['_' + type + 'Btn'];
234             if(btn){
235                 $D.hide(btn);
236             }
237         },
238         /**
239          * 隐藏气泡
240          */
241         hide: function(){
242             $D.hide(this._container); 
243             if(this.option.closeOnHide){
244                 this.close();
245             }
246         },
247         /**
248          * 关闭并注销气泡
249          */
250         close: function(){
251             if(!this._isClosed){
252                 this._isClosed = true;
253                 $E.off(this._closeBtn, 'click');
254                 $E.off(this._okBtn, 'click');
255                 $E.off(this._nextBtn, 'click');
256                 if(this._container.parentNode){
257                     this._container.parentNode.removeChild(this._container);
258                 }
259                 for (var p in this) {
260                     if (this.hasOwnProperty(p)) {
261                         delete this[p];
262                     }
263                 }
264             }else{
265                 J.warn('Trying to close a closed bubbleTip!', 'BubbleTip');
266             }
267         },
268         /**
269          * 指示是否已关闭气泡
270          */
271         isClose: function(){
272             return this._isClosed;
273         },
274         /**
275          * 返回一个内部使用的id
276          * @private
277          * @return {Int} 
278          */
279         _getId: function(){
280             if(!Bubble.__id){
281                 Bubble.__id = 0;
282             }
283             return Bubble.__id++;
284         },
285         /**
286          * 设置箭头的指示方向和位置
287          * @private
288          * @param {String} position 箭头的位置, 由两个单词(top/bottom/left/right中的一个)组成
289          * @param {Int} offset 箭头的偏移位置
290          * @example
291          *  position的值可为
292          *  "top left"
293          *  "top right"
294          *  "bottom left"
295          *  "bottom right"
296          *  "left top"
297          *  "left bottom"
298          *  "right top"
299          *  "right bottom"
300          */
301         _setPointerPosition: function(position, offset){
302             var posArr = position.split(' ');
303             var pointer = this._pointer;
304             $D.setClass(pointer, 'bubble_tip_pointer bubble_tip_pointer_' + posArr[0]);
305             $D.setCssText(pointer, '');
306             $D.setStyle(pointer, posArr[1], offset + 'px');
307         },
308         /**
309          * 设置气泡的位置
310          * @private
311          * @param {Object} option show 的参数
312          */
313         _setBubblePosition: function(option){
314             var position = option.position;//直接设置position后就不要设置target
315             
316             if(option.target){
317                 var posArr = option.pointerPosition.split(' ');
318                 var posObj = this._calculateBubblePosition(option.target, option.pointerSize, option.pointerOffset);
319                 var sub = 0;
320                 if(/top|bottom/.test(posArr[0])){
321                     sub = 1;
322                 }
323                 position[0] = posObj[posArr[sub] + sub];
324                 sub = (sub + 1) % 2;
325                 position[1] = posObj[posArr[sub] + sub];
326             }
327             var x = position[0] + option.targetOffset[0];
328             var y = position[1] + option.targetOffset[1];
329             $D.setStyle(this._container, 'left', x + 'px');
330             $D.setStyle(this._container, 'top', y + 'px');
331         },
332         /**
333          * 计算气泡的位置
334          * @param {HTMLElement} target 箭头所指的dom
335          * @param {Array} pointerSize 箭头的大小,[width, height]
336          * @param {Int} pointerOffset 箭头的偏移
337          * @private
338          * @return {Object} 计算得到的结果
339          * @example 
340          * 返回的结果格式为
341          * {
342          *       top0: tt + th + ph,
343          *       bottom0: tt - bh - ph,
344          *       left0: tl + tw + ph,
345          *       right0: tl - bw - ph,
346          *       
347          *       top1: tt + th / 2 - pw,
348          *       bottom1: tt + th / 2 - bw + pw,
349          *       left1: tl + tw / 2- pw,
350          *       right1: tl + tw / 2 - bw + pw
351          *   }
352          */
353         _calculateBubblePosition: function(target, pointerSize, pointerOffset){
354             var container = this._container,
355                 targetPos = $D.getClientXY(target);
356             var bw = $D.getOffsetWidth(container), bh = $D.getOffsetHeight(container),
357                 tw = $D.getWidth(target), th = $D.getHeight(target),
358                 tl = targetPos[0], tt = targetPos[1],
359                 pw = pointerOffset + (pointerSize[0] / 2), ph = pointerSize[1];
360             return {
361                 top0: tt + th + ph,
362                 bottom0: tt - bh - ph,
363                 left0: tl + tw + ph,
364                 right0: tl - bw - ph,
365                 
366                 top1: tt + th / 2 - pw,
367                 bottom1: tt + th / 2 - bh + pw,
368                 left1: tl + tw / 2- pw,
369                 right1: tl + tw / 2 - bw + pw
370             };
371         },
372         /**
373          * 检查箭头的位置参参数是否合法
374          * @private
375          * @param {String} position 位置参数
376          */
377         _checkPointerPosition: function(position){
378             var posArr = position.split(' ');
379             var lrRegex = /left|right/,
380                 tbRegex = /top|bottom/;
381             if(tbRegex.test(posArr[0]) && lrRegex.test(posArr[1])){
382                 return true;
383             }else if(lrRegex.test(posArr[0]) && tbRegex.test(posArr[1])){
384                 return true;
385             }
386             return false;
387         }
388         
389     });
