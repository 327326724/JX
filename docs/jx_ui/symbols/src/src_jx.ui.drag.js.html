 27 /**
 28  * 拖拽模块
 29  */
 30 Jx().$package(function (J) {
 31     var $D = J.dom,
 32         $E = J.event;
 33 
 34     var ieSelectFix = function (e) {
 35         e.preventDefault();
 36         //return false;
 37     };
 38     
 39     var _workAreaWidth=false,
 40         _workAreaHeight=false,
 41         _clientWidth,
 42         _clientHeight,
 43         _width=false,
 44         _height=false,
 45         _limit_right,
 46         _limit_left,
 47         _limit_top,
 48         _limit_bottom;
 49         
 50     //for ipad
 51     var _dragOffsetX,
 52         _dragOffsetY;
 53         
 54     var _dragMaskLayer,
 55         _maskLayerZIndex = 9000000;
 56     
 57     /**
 58      * 拖拽类
 59      * 
 60      * @memberOf ui
 61      * @class
 62      * @name Drag
 63      * @constructor
 64      * 
 65      * @param {Element} apperceiveEl 监听拖拽动作的元素
 66      * @param {Element} effectEl 展现拖拽结果的元素
 67      * @param {Object} option 其他选项，如:isLimited,leftMargin...
 68      * 
 69      * var limiteOption = {
 70             isLimited : 是否有界限,
 71             clientEl :盒子模型,边界用
 72             rightMargin : 边界距离,
 73             leftMargin : 边界距离,
 74             bottomMargin : 边界距离,
 75             topMargin : 边界距离,
 76             isOverLeft:
 77             isOverRight:
 78             isOverTop:
 79             isOverBottom:能否超出边界,支持拖出去
 80         };
 81      */
 82     J.ui = J.ui || {};
 83     J.ui.Drag = new J.Class(
 84     /**
 85      * @lends ui.Drag.prototype
 86      */    
 87     {
 88         /** @ignore */
 89         init: function (apperceiveEl, effectEl, option) {
 90             var context = this;
 91             var curDragElementX, curDragElementY, dragStartX, dragStartY;
 92             this.apperceiveEl = apperceiveEl;
 93             option = option || {};
 94             this.isLimited = option.isLimited || false;
 95             this.dragType = option.dragType;
 96             this.isLocked = option.isLocked || false;
 97             this.isLockCursorInScreen = option.isLockCursorInScreen || false;
 98             var isMoved = false;
 99             if (this.isLimited) {
100                 this._leftMargin = option.leftMargin || 0;
101                 this._topMargin = option.topMargin || 0;
102                 this._rightMargin = option.rightMargin || 0;
103                 this._bottomMargin = option.bottomMargin || 0;
104             }
105             if(option.xOnly){
106                 this._xOnly = option.xOnly||false;
107             }
108             if(option.yOnly){
109                 this._yOnly = option.yOnly||false;
110             }
111             //TODO for flex wmode-non-transparent
112 
113             if (effectEl === false) {
114                 this.effectEl = false;
115             } else {
116                 this.effectEl = effectEl || apperceiveEl;
117             }
118 
119             /** @ignore */
120             this.dragStart = function (e) {
121                 if(e.changedTouches){//多点触摸
122                     if(e.changedTouches.length>1){
123                         return;
124                     }
125                     e=e.changedTouches[0];
126                     document.body.style['WebkitTouchCallout']='none';
127                 }else{
128                     e.preventDefault();
129                     e.stopPropagation();
130                 }
131                 context.dragStartIn(e.pageX,e.pageY);
132                 
133             };
134             /** @ignore */
135             this.dragStartIn = function (x,y) {
136                 if(context.isLocked){
137                     return;
138                 }
139                 //缓存高宽
140                 $E.notifyObservers(context, "beforeStart");
141                 $E.notifyObservers(document, "beforeStart");
142 //              if(this.dragType){//jet框架跟alloy耦合不好
143 //                  $E.notifyObservers(alloy, "dragBeforeStart",this.dragType);
144 //              }
145                 _clientWidth = $D.getClientWidth(),
146                 _clientHeight = $D.getClientHeight();
147                 _workAreaWidth = option.clientEl?$D.getClientWidth(option.clientEl):_clientWidth;
148                 _workAreaHeight = option.clientEl?$D.getClientHeight(option.clientEl):_clientHeight;
149                 _width = context.effectEl ? parseInt($D.getClientWidth(context.effectEl)) : 0;
150                 _height = context.effectEl ? parseInt($D.getClientHeight(context.effectEl)) : 0;
151 
152                 if (context.isLimited) {
153                     _limit_right = _workAreaWidth - _width - context._rightMargin;
154                     _limit_left = context._leftMargin;
155                     _limit_top = context._topMargin;
156                     _limit_bottom = _workAreaHeight - _height - context._bottomMargin;
157                 }
158                 if(!_dragMaskLayer){
159                     _dragMaskLayer = new J.ui.MaskLayer({opacity: 0});
160 //                    _maskLayerZIndex = alloy.layout.getTopZIndex(2);
161                 }
162                 _dragMaskLayer.setZIndex(_maskLayerZIndex);
163                 _dragMaskLayer.show();
164 //                context._oldZIndex = $D.getStyle(context.effectEl, 'zIndex') || 0;
165 //                J.out('dragStart - _oldZIndex: ' + context._oldZIndex + '::' + _maskLayerZIndex);
166 //                $D.setStyle(context.effectEl, 'zIndex', _maskLayerZIndex + 1);
167                 
168                 
169                 context._oldX=curDragElementX = context.effectEl ? (parseInt($D.getStyle(context.effectEl, "left")) || 0) : 0;
170                 context._oldY=curDragElementY = context.effectEl ? (parseInt($D.getStyle(context.effectEl, "top")) || 0) : 0;
171                 dragStartX = x;
172                 dragStartY = y;
173                 if(J.browser.mobileSafari){
174                     $E.on(document, 'touchmove', context.dragMove);
175                     $E.on(document, 'touchend', context.dragStop);
176                     
177                     var oldMatrix=new WebKitCSSMatrix(window.getComputedStyle(context.apperceiveEl).webkitTransform);
178                     _dragOffsetX = x - oldMatrix.m41;
179                     _dragOffsetY = y - oldMatrix.m42;
180                     
181                 }else{
182                     $E.on(document, "mousemove", context.dragMove);
183                     $E.on(document, "mouseup", context.dragStop);
184                 }
185                 if (J.browser.ie) {
186                     $E.on(document.body, "selectstart", ieSelectFix);
187                 }
188                 if(J.browser.mobileSafari){
189                     $E.notifyObservers(context, "start", { x: x, y: y });
190                 }else{
191                     $E.notifyObservers(context, "start", { x: curDragElementX, y: curDragElementY });
192                 }      
193             };
194             /** @ignore */
195             this.dragMove = function (e) {
196                 if(context.isLocked){
197                     return;
198                 }
199                 if(e.browserEvent){
200                     e.browserEvent.preventDefault();
201                     e.browserEvent.stopPropagation();
202                 }else{
203                     e.preventDefault();
204                     e.stopPropagation();
205                 }
206                 if(e.changedTouches){//多点触摸
207                     
208                     e=e.changedTouches[0];
209                 }
210                 var x,y;
211                 var pageX = e.pageX,
212                     pageY = e.pageY;
213                 if(context.isLockCursorInScreen){//限制鼠标超出浏览器外
214                     pageX < 0 ? (pageX = 0) : (pageX > _clientWidth ? (pageX = _clientWidth) : 0);
215                     pageY < 0 ? (pageY = 0) : (pageY > _clientHeight ? (pageY = _clientHeight) : 0);
216                 }
217                 if(!J.browser.mobileSafari){
218                     x = curDragElementX + (pageX - dragStartX);
219                     y = curDragElementY + (pageY - dragStartY);
220                 }
221                 if (context.isLimited) {
222                     if(x>_limit_right &&!option.isOverRight ){
223                         x=_limit_right;
224                     }
225                     if(x<_limit_left && !option.isOverLeft){
226                         x=_limit_left;
227                     }
228                 }
229                 if (context._oldX !== x&&!context._yOnly) {
230                     context._oldX = x;
231                     if (context.effectEl && !J.browser.mobileSafari) {
232                         context.effectEl.style.left = x + "px";
233                     }
234                     isMoved = true;
235                 }
236                 
237                 //J.out("context._topMargin: "+context._topMargin);
238                 if (context.isLimited) {
239                     if (y > _limit_bottom && !option.isOverBottom) {
240                         y = _limit_bottom;
241                     }
242                     if (y < _limit_top && !option.isOverTop) {
243                         y = _limit_top;
244                     }
245                 }
246 
247                 if (context._oldY !== y&&!context._xOnly) {
248                     context._oldY = y;
249                     if (context.effectEl && !J.browser.mobileSafari) {
250                         context.effectEl.style.top = y + "px";
251                     }
252                     isMoved = true;
253                 }
254                 
255                 //for ipad...BAD SMELL.
256                 var notifyX = x,
257                     notifyY = y;
258                 if(context.effectEl && J.browser.mobileSafari){
259                     context._oldX = curDragElementX + (pageX - dragStartX);
260                     context._oldY = curDragElementY + (pageY - dragStartY);
261                     var eX = pageX,
262                         eY = pageY;
263                     if(!context._yOnly){
264                         x = pageX - _dragOffsetX;
265                     }else{
266                         x = curDragElementX;
267                     }
268                     if(!context._xOnly){
269                         y = eY - _dragOffsetY;
270                     }else{
271                         y = curDragElementY;
272                     }
273                     context.effectEl.style.webkitTransform = 'translate3d(' + x + 'px, ' + y + 'px, 0px)';
274                     notifyX = pageX;
275                     notifyY = pageY;
276                 }
277 
278 
279                 if (isMoved) {
280                     $E.notifyObservers(context, "move", { x: notifyX, y: notifyY, orientEvent:e });
281                 }
282 
283             };
284             /** @ignore */
285             this.dragStop = function (e) {
286                 _dragMaskLayer.hide();
287                 if(context.isLocked){
288                     return;
289                 }
290 //                J.out('dragStop - _oldZIndex: ' + context._oldZIndex);
291 //                $D.setStyle(context.effectEl, 'zIndex', context._oldZIndex);
292                 
293                 document.body.style['WebkitTouchCallout']='auto';
294                 if(isMoved || J.browser.mobileSafari) {
295                     var x=context._oldX;
296                     var y=context._oldY;
297                     if (context.isLimited) {
298                         if(x>_limit_right&&!option.isOverRight){
299                             x=_limit_right;
300                         }
301                         if(x<_limit_left&&!option.isOverLeft){
302                             x=_limit_left;
303                         }
304                     }
305                     if (context.isLimited) {
306                         if (y > _limit_bottom&&!option.isOverBottom) {
307                             y = _limit_bottom;
308                         }
309                         if (y < _limit_top&&!option.isOverTop) {
310                             y = _limit_top;
311                         }
312                     }
313                     if(J.browser.mobileSafari){
314                         e.preventDefault();
315                         if(option.noEndCallback && context.effectEl){
316                             context.effectEl.style.webkitTransform  = 'none';
317                             $D.setStyle(context.effectEl,'left',x+'px');
318                             $D.setStyle(context.effectEl,'top',y+'px');
319                         }
320                         $E.notifyObservers(context, "end", { x: x, y: y, orientEvent:e.changedTouches[0] });
321                     }else{
322                         $E.notifyObservers(context, "end", { x: x, y: y, orientEvent:e });
323                     }
324                 }
325                 else {
326                     $E.notifyObservers(context, "end", {orientEvent:e});
327                 }
328                 if (context.isLimited&&(context.isOverRight||context.isOverLeft||context.isOverTop||context.isOverBottom)) {
329                     var x = curDragElementX + (e.pageX - dragStartX);
330                     var y = curDragElementY + (e.pageY - dragStartY);
331                     var tempR = _workAreaWidth - _width - context._rightMargin;
332                     var tempL = context._leftMargin;
333                     var tempB = _workAreaHeight - context._bottomMargin;
334                     var tempT = context._topMargin;
335                     if (x > tempR||x < tempL||y > tempB||y < tempT) {
336                         //超出边界
337                         $E.notifyObservers(context,"overFlowBorder",{x: x,y: y});
338                         //J.out("overFlow");
339                     }
340                 }
341                 _workAreaWidth = false;
342                 _workAreaHeight = false;
343                 _width = false;
344                 _height = false;
345                 if(J.browser.mobileSafari){
346                     $E.off(document, 'touchmove', context.dragMove);
347                     $E.off(document, 'touchend', context.dragStop);
348                 }else{
349                     $E.off(document, "mousemove", context.dragMove);
350                     $E.off(document, "mouseup", context.dragStop);
351                 }
352                 if (J.browser.ie) {
353                     $E.off(document.body, "selectstart", ieSelectFix);
354                 }
355                 isMoved= false;
356                 //J.out("end")
357             };
358 
359             //if(J.browser.mobileSafari){
360             //  $E.on(this.apperceiveEl, "touchstart", this.dragStart);
361             //}else{
362                 $E.on(this.apperceiveEl, "drag", this.dragStart);
363             //}
364         },
365         /**
366          * 设置展现拖拽结果的元素
367          * @param {HTMLElement} el
368          */
369         setEffect: function(el){
370             this.effectEl = el;
371         },
372         /**
373          * 锁定拖拽
374          */
375         lock: function () {
376             //if(J.browser.mobileSafari){
377             //  $E.off(this.apperceiveEl, "touchstart", this.dragStart);
378             //}else{
379                 this.isLocked = true;
380                 $E.off(this.apperceiveEl, "drag", this.dragStart);
381             //}
382         },
383         /**
384          * 解锁
385          */
386         unlock: function () {
387             //if(J.platform.iPad){/
388             //  $E.on(this.apperceiveEl, "touchstart", this.dragStart);
389             //}else{
390                 this.isLocked = false;
391                 $E.on(this.apperceiveEl, "drag", this.dragStart);
392             //}
393         },
394         /**
395          * 显示监听拖拽动作的元素
396          */
397         show: function () {
398             $D.show(this.apperceiveEl);
399         },
400         /**
401          * 隐藏监听拖拽动作的元素
402          */
403         hide: function () {
404             $D.hide(this.apperceiveEl);
405         },
406         /**
407          * 设置拖拽边界
408          */
409         setLimite : function(option){
410             option = option || {};
411             this.isLimited = option.isLimited || true;
412             if (this.isLimited) {
413                 this._leftMargin = option.leftMargin || 0;
414                 this._topMargin = option.topMargin || 0;
415                 this._rightMargin = option.rightMargin || 0;
416                 this._bottomMargin = option.bottomMargin || 0;
417             }
418         }
419     });
